"""
==========================================================================
SYSTEM_SERVICES - SYSTEM_LOGGER (LOG ASÍNCRONO SEGURO CON MULTI-LISTENER)
==========================================================================
"""
import logging
import logging.handlers
import queue
import sys
import threading
from datetime import datetime
from pathlib import Path
from typing import Dict, Any

from SYSTEM_SERVICES.system_paths import (
    LOGS_MASTER_DIR,
    LOGS_CAPTURE_DIR,
    LOGS_AUDIT_DIR,
    LOGS_QUANT_DIR,
    LOGS_DECISION_DIR
)

CATEGORY_MAP: Dict[str, Path] = {
    "00_ORCHESTRATOR": LOGS_MASTER_DIR,
    "01_CAPTURE": LOGS_CAPTURE_DIR,
    "02_AUDIT": LOGS_AUDIT_DIR,
    "03_QUANT": LOGS_QUANT_DIR,
    "04_DECISION": LOGS_DECISION_DIR
}

# Control de inicialización concurrente de loggers
_logger_lock = threading.Lock()
_active_listeners: Dict[str, logging.handlers.QueueListener] = {}


def get_system_logger(module_name: str, category: str = "00_ORCHESTRATOR") -> logging.Logger:
    """
    Genera y retorna un logger asíncrono y robusto para cualquier motor del bot,
    garantizando que sus logs se mantengan estrictamente en su archivo correspondiente.
    """
    with _logger_lock:
        logger = logging.getLogger(module_name)
        
        # Si el logger ya tiene configurado un QueueHandler, evitamos duplicar handlers
        if any(isinstance(h, logging.handlers.QueueHandler) for h in logger.handlers):
            return logger
            
        logger.setLevel(logging.INFO)
        logger.propagate = False  # Evita duplicación hacia el logger raíz
        
        # Resolver ruta destino
        log_folder = CATEGORY_MAP.get(category, LOGS_MASTER_DIR)
        current_date = datetime.now().strftime("%Y%m%d")
        log_file_path = log_folder / f"{module_name}_{current_date}.log"
        
        # Formato unificado
        formatter = logging.Formatter(
            "[%(asctime)s.%(msecs)03d] [%(levelname)s] [%(name)s] -> %(message)s",
            datefmt="%Y-%m-%d %H:%M:%S"
        )
        
        # Handlers físicos de destino
        file_handler = logging.handlers.RotatingFileHandler(
            log_file_path,
            maxBytes=10 * 1024 * 1024,
            backupCount=5,
            encoding="utf-8"
        )
        file_handler.setFormatter(formatter)
        
        console_handler = logging.StreamHandler(sys.stdout)
        console_handler.setFormatter(formatter)
        
        # Cola y Listener dedicado exclusivo para ESTE módulo
        module_queue = queue.Queue(-1)
        queue_handler = logging.handlers.QueueHandler(module_queue)
        logger.addHandler(queue_handler)
        
        # Instanciar e iniciar listener para el módulo actual
        listener = logging.handlers.QueueListener(module_queue, file_handler, console_handler)
        listener.start()
        
        # Almacenar referencia al listener para poder cerrarlo en el shutdown
        _active_listeners[module_name] = listener
        
        return logger


def stop_logger_listener() -> None:
    """
    Detiene de forma segura todos los hilos listener de fondo del logger al apagar el bot.
    """
    with _logger_lock:
        for module_name, listener in list(_active_listeners.items()):
            try:
                listener.stop()
            except Exception:
                pass
        _active_listeners.clear()
