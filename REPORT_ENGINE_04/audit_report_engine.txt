import json
import re
from pathlib import Path
from typing import Optional, Dict, Any

# Importamos la infraestructura y utilidades necesarias (Corregido a QUANT_OUTPUTS)
from SYSTEM_SERVICES.system_paths import QUANT_OUTPUTS
from SYSTEM_SERVICES.system_settings import send_telegram_alert
from REPORT_ENGINE_04.ai_gateway import call_llm


def load_json_file(filepath: Path) -> Dict[str, Any]:
    """Carga un archivo JSON de forma segura."""
    if not filepath.exists():
        return {}
    try:
        with open(filepath, "r", encoding="utf-8") as f:
            return json.load(f)
    except Exception as e:
        print(f"[REPORT_ENGINE] Error leyendo JSON {filepath.name}: {e}")
        return {}


def detect_latest_date_in_folder() -> Optional[str]:
    """
    Escanea la carpeta de salidas cuantitativas para detectar la fecha
    del certificado cuantitativo más reciente en disco.
    Busca patrones de nombre: quant_certificate_YYYYMMDD.json
    """
    pattern = re.compile(r"quant_certificate_(\d{8})\.json")
    detected_dates = []
    
    for file in QUANT_OUTPUTS.glob("quant_certificate_*.json"):
        match = pattern.match(file.name)
        if match:
            detected_dates.append(match.group(1))
            
    if not detected_dates:
        return None
        
    # Retorna la fecha más reciente encontrada (ordenación de strings de fecha funciona con YYYYMMDD)
    return sorted(detected_dates)[-1]


def generate_session_report(fecha: Optional[str] = None) -> bool:
    """
    Ejecuta el ciclo de reporte completo:
      1. Detecta o recibe la fecha de análisis.
      2. Carga los JSON reales de quant_certificate y global_regime [1.1].
      3. Compila el prompt dinámico con la estructura de datos cruda.
      4. Consulta a la IA para el análisis cualitativo [1.1].
      5. Guarda el diagnóstico final en formato .md en C:\\MT5_DATA [1.1].
      6. Despacha una alerta con el resumen ejecutivo a Telegram.
    """
    # 1. Resolver fecha de análisis
    if not fecha:
        fecha = detect_latest_date_in_folder()
        if not fecha:
            print(f"[REPORT_ENGINE] ❌ No se encontraron archivos de datos ('quant_certificate_*.json') "
                  f"en la ruta: {QUANT_OUTPUTS}")
            return False
            
    print(f"[REPORT_ENGINE] 🔄 Procesando reporte real para la sesión: {fecha}")
    
    # 2. Rutas físicas de sus archivos reales de producción
    cert_path = QUANT_OUTPUTS / f"quant_certificate_{fecha}.json"
    regime_path = QUANT_OUTPUTS / f"global_regime_{fecha}.json"
    report_output_path = QUANT_OUTPUTS / f"diagnostico_operacional_F3_{fecha}.md"
    
    # 3. Cargar datos duros reales
    cert_data = load_json_file(cert_path)
    regime_data = load_json_file(regime_path)
    
    if not cert_data and not regime_data:
        print(f"[REPORT_ENGINE] ❌ No se pudieron cargar los datos para la fecha {fecha}. "
              f"Verifique que existan los archivos en {QUANT_OUTPUTS}")
        return False

    # 4. Compilar el prompt con los datos reales en formato JSON indentado
    prompt = (
        f"Analiza los siguientes datos operativos reales de la sesión de trading {fecha}.\n"
        f"Evalúa la aptitud de los activos, anomalías y clasifica la sesión técnica de hoy.\n\n"
        f"### 1. CERTIFICADO CUANTITATIVO DE APTITUD (quant_certificate):\n"
        f"```json\n{json.dumps(cert_data, indent=2)}\n```\n\n"
        f"### 2. COMPORTAMIENTO Y REGÍMENES DE MERCADO (global_regime):\n"
        f"```json\n{json.dumps(regime_data, indent=2)}\n```\n\n"
        f"Redacta un diagnóstico operacional técnico, detallado y fácil de leer. "
        f"Usa Markdown estructurado con títulos, viñetas y tablas si es conveniente. "
        f"Al final, añade una sección con el título '### RESUMEN EJECUTIVO TELEGRAM' de "
        f"máximo un párrafo corto que resuma las decisiones operativas tomadas hoy."
    )
    
    system_instruction = (
        "Eres un analista cuantitativo jefe de una mesa de trading de alta frecuencia. "
        "Tu trabajo es traducir datos de sistemas numéricos a conclusiones ejecutivas legibles "
        "y precisas para el director del fondo de inversión."
    )

    # 5. Consultar a la IA (DeepSeek / OpenAI)
    print("[REPORT_ENGINE] 🧠 Enviando datos de producción a la IA para diagnóstico cualitativo...")
    analisis_cualitativo = call_llm(prompt, system_instruction)
    
    # 6. Guardar el reporte en formato Markdown (.md) en disco de forma segura
    try:
        with open(report_output_path, "w", encoding="utf-8") as f:
            f.write(analisis_cualitativo)
        print(f"[REPORT_ENGINE] ✅ Reporte Markdown guardado con éxito en: {report_output_path}")
    except Exception as e:
        print(f"[REPORT_ENGINE] ❌ Error escribiendo archivo de reporte final: {e}")
        return False
        
    # 7. Extraer el resumen para Telegram y despacharlo
    telegram_text = ""
    match = re.search(r"### RESUMEN EJECUTIVO TELEGRAM\s*(.*)", analisis_cualitativo, re.DOTALL)
    if match:
        telegram_text = match.group(1).strip()
    else:
        telegram_text = f"Análisis de sesión {fecha} completado. Reporte guardado en disco."

    # Enviar alerta asíncrona
    mensaje_alerta = f"🌙 *DIAGNÓSTICO DE SESIÓN — {fecha}*\n\n{telegram_text}"
    send_telegram_alert(mensaje_alerta)
    print("[REPORT_ENGINE] 📱 Resumen ejecutivo enviado a Telegram de forma asíncrona.")
    
    return True


if __name__ == "__main__":
