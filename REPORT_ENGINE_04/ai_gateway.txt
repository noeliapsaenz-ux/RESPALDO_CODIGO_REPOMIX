import json
import urllib.request
import urllib.parse
from typing import Dict, Any, Optional, Callable, Tuple

from SYSTEM_SERVICES.system_settings import (
    DEEPSEEK_API_KEY,
    OPENAI_API_KEY,
    QWEN_API_KEY,
    GEMINI_API_KEY,
    ANTHROPIC_API_KEY,
    AI_PROVIDER
)

# ==========================================================================
# ADAPTADORES POR FAMILIA DE API (build_request / parse_response)
# ==========================================================================
# Cada adaptador recibe (api_key, model, system_instruction, prompt, url)
# y retorna (url_final, headers, payload). El parser recibe el JSON crudo
# de respuesta y retorna únicamente el texto generado.

def _build_openai_compatible(api_key: str, model: str, system_instruction: str,
                              prompt: str, url: str) -> Tuple[str, Dict[str, str], Dict[str, Any]]:
    """Familia OpenAI-compatible: DeepSeek, OpenAI, Qwen (modo compatible de DashScope)."""
    headers = {
        "Authorization": f"Bearer {api_key}",
        "Content-Type": "application/json"
    }
    payload = {
        "model": model,
        "messages": [
            {"role": "system", "content": system_instruction},
            {"role": "user", "content": prompt}
        ],
        "temperature": 0.2
    }
    return url, headers, payload

def _parse_openai_compatible(response_json: Dict[str, Any]) -> str:
    return response_json["choices"][0]["message"]["content"]

def _build_gemini(api_key: str, model: str, system_instruction: str,
                   prompt: str, url: str) -> Tuple[str, Dict[str, str], Dict[str, Any]]:
    """Gemini: la API key va como query param, no como header; esquema contents/parts."""
    final_url = f"{url}/{model}:generateContent?key={api_key}"
    headers = {"Content-Type": "application/json"}
    payload = {
        "contents": [
            {"role": "user", "parts": [{"text": prompt}]}
        ],
        "systemInstruction": {"parts": [{"text": system_instruction}]},
        "generationConfig": {"temperature": 0.2}
    }
    return final_url, headers, payload

def _parse_gemini(response_json: Dict[str, Any]) -> str:
    return response_json["candidates"][0]["content"]["parts"][0]["text"]

def _build_anthropic(api_key: str, model: str, system_instruction: str,
                      prompt: str, url: str) -> Tuple[str, Dict[str, str], Dict[str, Any]]:
    """Claude (Anthropic Messages API): system es campo top-level, no un mensaje más."""
    headers = {
        "x-api-key": api_key,
        "anthropic-version": "2023-06-01",
        "Content-Type": "application/json"
    }
    payload = {
        "model": model,
        "max_tokens": 1024,
        "system": system_instruction,
        "messages": [{"role": "user", "content": prompt}],
        "temperature": 0.2
    }
    return url, headers, payload

def _parse_anthropic(response_json: Dict[str, Any]) -> str:
    return response_json["content"][0]["text"]

# REGISTRO DE PROVEEDORES (configuración + API key + adaptador)
# ==========================================================================
_PROVIDERS_CONFIG: Dict[str, Dict[str, Any]] = {
    "DEEPSEEK": {

        "url": "https://api.deepseek.com/chat/completions",

        "model": "claude-sonnet-4-6",
        "build_request": _build_anthropic,
        "parse_response": _parse_anthropic,
    },
}

_PRIORITY_ORDER = ["DEEPSEEK", "OPENAI", "QWEN", "GEMINI", "CLAUDE"]

def _select_provider() -> Optional[str]:
    """
    Resuelve qué proveedor usar.
    - Si AI_PROVIDER fuerza uno explícito: lo usa solo si tiene API key cargada;
      si no la tiene, falla explícito (no cae en otro proveedor por sorpresa).
    - Si AI_PROVIDER == "AUTO": usa el primero disponible según _PRIORITY_ORDER
      (hoy en la práctica siempre será DEEPSEEK, porque es el único con key).
    """
    if AI_PROVIDER != "AUTO":
        cfg = _PROVIDERS_CONFIG.get(AI_PROVIDER)
        if cfg and cfg["api_key"]:
            return AI_PROVIDER
        return None

    for name in _PRIORITY_ORDER:
        if _PROVIDERS_CONFIG[name]["api_key"]:
            return name
    return None

def call_llm(prompt: str, system_instruction: str = "Eres un analista experto en trading algorítmico.") -> str:
    """
    Realiza una consulta a la API de Inteligencia Artificial seleccionada.
    El proveedor se resuelve vía AI_PROVIDER (.env) o automáticamente según
    qué API key esté presente, en el orden DEEPSEEK > OPENAI > QWEN > GEMINI > CLAUDE.

    Args:
        prompt: El contenido del análisis/datos que se enviarán al modelo.
        system_instruction: Instrucciones de rol de comportamiento del modelo.

    Returns:
        str: Respuesta cualitativa del modelo de lenguaje en formato Markdown o texto.
    """
    provider_name = _select_provider()

    if not provider_name:
        if AI_PROVIDER != "AUTO":
            return (
                f"⚠️ [AI_GATEWAY] AI_PROVIDER está fijado en '{AI_PROVIDER}' pero no se "
                f"detectó su API key correspondiente en el .env."
            )
        return (
            "⚠️ [AI_GATEWAY] No se pudo realizar el análisis cualitativo. "
            "No se detectó ninguna API key de IA (DEEPSEEK, OPENAI, QWEN, GEMINI o CLAUDE) "
            "en el archivo de configuración .env."
        )

    config = _PROVIDERS_CONFIG[provider_name]

    try:
        url, headers, payload = config["build_request"](
            config["api_key"], config["model"], system_instruction, prompt, config["url"]
        )
        data_bytes = json.dumps(payload).encode("utf-8")
        req = urllib.request.Request(url, data=data_bytes, headers=headers, method="POST")

        with urllib.request.urlopen(req, timeout=30) as response:
            res_data = json.loads(response.read().decode("utf-8"))

        return config["parse_response"](res_data)

    except Exception as e:
        return f"⚠️ [AI_GATEWAY] Fallo en la llamada HTTP de red al proveedor {provider_name}: {e}"
