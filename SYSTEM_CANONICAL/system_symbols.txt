==========================================================================
SYSTEM_CANONICAL - SYSTEM_SYMBOLS (AUTORIDAD DE SÍMBOLOS DEFINITIVA)

Ubicación: SYSTEM_CANONICAL/system_symbols.py
Responsabilidad: Catálogo oficial e inmutable de los 10 pares Forex autorizados.
                 - No se permite ningún par adicional.
                 - Libre de dependencias externas.

POLÍTICA DE ERRORES (DECISIÓN DE DISEÑO):
   - Se implementa el principio de "Fail-Fast" (Fallo Inmediato).
   - Consultar o validar un activo que no pertenezca a la tupla oficial
     lanzará un KeyError de inmediato.
   - Esto evita degradaciones silenciosas que podrían corromper los cálculos
     del motor cuantitativo o de toma de decisiones en producción.
==========================================================================
"""
from dataclasses import dataclass
from typing import Dict, Tuple


@dataclass(frozen=True)
class SymbolSpecification:
    """
    Especificación técnica e inmutable de un activo financiero.
    """
    symbol: str              # Nombre oficial del símbolo (ej. "EURUSD")
    digits: int              # Decimales (5 para Forex estándar, 3 para JPY)
    spread_limit: float      # Límite de spread permitido en puntos enteros del broker
    buffer_size: int         # Tamaño del búfer en memoria RAM
    min_buffer_ready: int    # Ticks mínimos requeridos para habilitar análisis
    is_risk_asset: bool      # Clasificación de activo de riesgo (F3 / Quant)
    is_safe_haven: bool      # Clasificación de activo de refugio seguro (F3 / Quant)
    backpressure_multiplier: float  # Multiplicador canónico para control de flujo

# ==========================================================================
# INTERFAZ DE CONSULTA DE LA CAPA CANÓNICA
# ==========================================================================

def get_symbols() -> Tuple[str, ...]:
    """
    Retorna la tupla inmutable con los 10 pares autorizados en producción.
    """
    return SYMBOLS


def is_symbol_valid(symbol: str) -> bool:
    """
    Verifica si un símbolo pertenece al conjunto autorizado de 10 pares.
    """
    return symbol.upper() in SYMBOLS


def get_symbol_specification(symbol: str) -> SymbolSpecification:
    """
    Retorna las especificaciones técnicas e inmutables del activo.

    POLÍTICA DE ERRORES:
       - Si el símbolo consultado no existe en el catálogo, se lanza un KeyError
         de forma inmediata para advertir al desarrollador del fallo.
    """
    sym_upper = symbol.upper()
    if sym_upper not in _CANONICAL_CATALOG:
        raise KeyError(
            f"[SYSTEM_SYMBOLS] ❌ Error de violación canónica: El activo '{symbol}' "
            f"no está registrado en el catálogo de producción de {len(SYMBOLS)} símbolos."
        )
    return _CANONICAL_CATALOG[sym_upper]


def get_spread_limit(symbol: str) -> float:
    """
    Retorna el límite de spread canónico para un símbolo.
    No admite límites 'hardcoded' locales. Fallback defensivo de 30.0 puntos.
    """
    try:
        return get_symbol_specification(symbol).spread_limit
    except KeyError:
        return 30.0
