# Inicializar logger
logger = init_audit_logging("audit_quality_trend_analyzer", "02_AUDIT")


def analizar_tendencias_calidad(ventana_dias: int = 5) -> Dict[str, Any]:
    """
    Analiza las tendencias de calidad de los datos históricos.

    Args:
        ventana_dias: Número de días a analizar

    Returns:
        Dict[str, Any]: Diccionario con tendencias por símbolo
    """
    history_file = AUDIT_STATE_DIR / "confidence_history.parquet"
    if not history_file.exists():
        return {"error": "No se localizó el historial de estabilidad unificado."}

    try:
        df = pd.read_parquet(history_file)
    except Exception as e:
        return {"error": f"Error leyendo base de datos de auditoría: {str(e)}"}

    if df.empty or len(df) < 2:
        return {"error": "Muestras estadísticas insuficientes."}

    df = df.sort_values(by=["symbol", "fecha"])
    diagnosticos = {}

    for symbol in df["symbol"].unique():
        df_sym = df[df["symbol"] == symbol].tail(ventana_dias)
        if len(df_sym) < 3:
            continue

        ingestion_scores = df_sym["capture_conf"].values
        overall_scores = df_sym["overall_conf"].values

        x = np.arange(len(overall_scores))
        slope, _ = np.polyfit(x, overall_scores, 1)
        slope = float(slope)

        direccion = "MEJORANDO" if slope > 1.5 else ("DETERIORANDO" if slope < -1.5 else "ESTABLE")
        caida_consecutiva = bool(overall_scores[-1] < overall_scores[-2] < overall_scores[-3]) if len(overall_scores) >= 3 else False

        diagnosticos[symbol] = {
            "symbol": symbol,
            "ultima_fecha": str(df_sym["fecha"].values[-1]),
            "ultima_confianza_ingesta": float(ingestion_scores[-1]),
            "ultima_confianza_consolidada": float(overall_scores[-1]),
            "historico_consolidado": [float(c) for c in overall_scores],
            "slope_calidad": round(slope, 3),
            "estado_tendencia": direccion,
            "alerta_degradacion_consecutiva": caida_consecutiva
        }

    return diagnosticos


def main():
    parser = argparse.ArgumentParser(description="Audit Quality Trend Analyzer")
    parser.add_argument("--date", type=str, default="latest")
    parser.add_argument("--window", "-w", type=int, default=5)
    args = parser.parse_args()

    fecha = resolve_target_date(args.date, AUDIT_STATE_DIR, get_fecha_broker_str)
    logger.info(f"📊 ANALIZADOR HISTÓRICO DE ESTABILIDAD - FECHA: {fecha}")

    tendencias = analizar_tendencias_calidad(ventana_dias=max(3, args.window))

    output_path = AUDIT_REPORTES_DIR / f"memoria_tendencias_{fecha}.json"
    with open(output_path, 'w', encoding='utf-8') as f:
        json.dump({"fecha": fecha, "tendencias": tendencias}, f, indent=2, ensure_ascii=False)
    logger.info(f"💾 Memoria de estabilidad guardada en: {output_path.name}")


if __name__ == "__main__":
    main()
