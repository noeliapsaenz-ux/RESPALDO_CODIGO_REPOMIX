def contar_ticks_jsonl(file_path: Path) -> int:
    """Cuenta de manera eficiente el número de líneas (ticks) en un archivo JSONL/GZ."""
    try:
        open_func = gzip.open(file_path, "rt", encoding="utf-8", errors="ignore") if file_path.suffix == ".gz" else open(file_path, "r", encoding="utf-8", errors="ignore")
        with open_func as f:
            return sum(1 for _ in f)
    except Exception:
        return 0

def calcular_hash_contenido_jsonl(file_path: Path) -> str:
    """Calcula el hash SHA-256 de las líneas de datos útiles del archivo."""
    if not file_path.exists():
        return ""
    try:
        sha = hashlib.sha256()
        open_func = gzip.open(file_path, "rt", encoding="utf-8", errors="ignore") if file_path.suffix == ".gz" else open(file_path, "r", encoding="utf-8", errors="ignore")
        with open_func as f:
            for line in f:
                if line.strip():
                    sha.update(line.strip().encode('utf-8'))
        return sha.hexdigest()
    except Exception:
        return ""

def obtener_metadatos_jsonl(symbol: str, fecha: str) -> Dict[str, Any]:
    """Recopila la metadata física del archivo de ticks original."""
    symbol = symbol.upper()

    search_paths = [
        PRIMARY_TICK_DIR / symbol / fecha,
        SSD_EVIDENCE_DIR / symbol / fecha,
        SAFE_DIR / symbol / fecha
    ]

    all_files = []
    for path in search_paths:
        if path.exists():
            all_files.extend(list(path.glob("*.jsonl")) + list(path.glob("*.jsonl.gz")))

    if not all_files:
        return {"exists": False, "total_ticks": 0}

    file_hashes = {}
    file_metadata = {}
    total_ticks = 0

    for fp in all_files:
        try:
            ticks = contar_ticks_jsonl(fp)
            file_metadata[fp.name] = {
                "size_bytes": fp.stat().st_size,
                "ticks": ticks,
                "is_gz": fp.suffix == ".gz"
            }
            total_ticks += ticks
            file_hashes[fp.name] = calcular_hash_contenido_jsonl(fp) or "ERROR"
        except Exception:
            pass

    content_hashes = [file_hashes[f.name] for f in sorted(all_files) if file_hashes.get(f.name) and file_hashes[f.name] != "ERROR"]
    combined_hash = hashlib.sha256("".join(content_hashes).encode()).hexdigest() if content_hashes else ""

    return {
        "exists": True,
        "total_ticks": total_ticks,
        "sha256_collective": combined_hash,
        "file_hashes": file_hashes,
        "file_metadata": file_metadata
    }

def obtener_metadatos_parquet(symbol: str, fecha: str) -> Dict[str, Any]:
    """Obtiene y valida la metadata estructural del Parquet Maestro."""
    parquet_path = AUDIT_MAESTROS / f"ticks_maestro_{symbol}_{fecha}.parquet"
    if not parquet_path.exists():
        return {"exists": False, "total_ticks": 0}
    try:
        df = pd.read_parquet(parquet_path)
        return {
            "exists": True,
            "total_ticks": len(df),
            "structure_valid": all(col in df.columns for col in COLUMNAS_REQUERIDAS_MAESTRO),
            "metadata": {
                "first_seq": int(df['seq'].min()) if 'seq' in df.columns and len(df) > 0 else None,
                "last_seq": int(df['seq'].max()) if 'seq' in df.columns and len(df) > 0 else None
            }
        }
    except Exception as e:
        return {"exists": False, "error": str(e), "total_ticks": 0}

def obtener_metadatos_gold(symbol: str, fecha: str) -> Dict[str, Any]:
    """Obtiene y valida la metadata estructural del Parquet Gold."""
    gold_path = AUDIT_GOLD / f"{symbol}_{fecha}_gold.parquet"
    if not gold_path.exists():
        return {"exists": False, "total_ticks": 0}
    try:
        df = pd.read_parquet(gold_path)
        return {
            "exists": True,
            "total_ticks": len(df),
            "metadata": {
                "first_seq": int(df['seq'].min()) if 'seq' in df.columns and len(df) > 0 else None,
                "last_seq": int(df['seq'].max()) if 'seq' in df.columns and len(df) > 0 else None
            }
        }
    except Exception:
        return {"exists": False, "total_ticks": 0}

def auditar_lineage(symbol: str, fecha: str) -> Dict[str, Any]:
    """Audita y compara las firmas hash para asegurar la inmutabilidad física."""
    symbol = symbol.upper()
    result = create_audit_result_template(symbol, stage="cryptographic_lineage")

    jsonl_meta = obtener_metadatos_jsonl(symbol, fecha)
    parquet_meta = obtener_metadatos_parquet(symbol, fecha)
    gold_meta = obtener_metadatos_gold(symbol, fecha)

    result["metadata"] = {
        "fecha": fecha,
        "calculated_at": datetime.now(timezone.utc).isoformat(),
        "stages": {
            "jsonl": jsonl_meta,
            "parquet": parquet_meta,
            "gold": gold_meta
        }
    }

    penalty = 0.0

    if not jsonl_meta.get("exists", False) or not parquet_meta.get("exists", False):
        result["status"] = "FAIL"
        result["confidence"] = 0.0
        result["critical_errors"].append({
            "code": "missing_mandatory_stages",
            "message": f"Faltan etapas físicas mandatorias del flujo para el símbolo {symbol}."
        })
        return result

    collective_hash = jsonl_meta.get("sha256_collective", "")
    state_file = AUDIT_STATE / "lineage_state.json"
    state = load_contract_safe(state_file)

    if symbol not in state:
        state[symbol] = {}

    if fecha in state[symbol]:
        if collective_hash != state[symbol][fecha]:
            result["critical_errors"].append({
                "code": "cryptographic_tamper_detected",
                "message": "Fallo de cadena de custodia: El hash colectivo actual no coincide con la firma inicial."
            })
            penalty += 50.0
    else:
        state[symbol][fecha] = collective_hash
        save_contract_atomic(state, state_file)

    if not parquet_meta.get("structure_valid", False):
        result["critical_errors"].append({
            "code": "invalid_structure",
            "message": "Esquema del Parquet Maestro no coincide con las columnas requeridas canónicas."
        })
        penalty += 25.0

    result["confidence"] = max(0.0, 100.0 - penalty)
    
    # Sincronización de umbrales canónicos para determinar el estatus de auditoría
    min_confidence = AUDIT_THRESHOLDS["confidence_minima"]
    warn_confidence = AUDIT_THRESHOLDS["completitud_minima"]
    
    result["status"] = "FAIL" if (result["critical_errors"] or result["confidence"] < min_confidence) else ("WARNING" if result["confidence"] < warn_confidence else "PASS")

    assert validate_audit_result(result)
    return result

def main():
    parser = argparse.ArgumentParser(description="Crypto Lineage Auditor")
    parser.add_argument("--date", type=str, default="latest")

    symbols = get_symbols()
    resultados = {s.upper(): auditar_lineage(s, fecha) for s in symbols}
    reporte = {
        "fecha": fecha,
        "timestamp_generacion": datetime.now(timezone.utc).isoformat(),
        "resultados": resultados
    }

    output_path = AUDIT_LINEAGE / f"lineage_report_{fecha}.json"
    save_contract_atomic(reporte, output_path)
    logger.info(f"✅ Reporte de linaje de datos emitido: {output_path.name}")

if __name__ == "__main__":
    main()
