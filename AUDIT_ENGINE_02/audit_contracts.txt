def load_json_safe(filepath: Path) -> Dict[str, Any]:
    """Carga controlada de JSON previniendo excepciones."""
    if not filepath.exists():
        return {}
    try:
        with open(filepath, "r", encoding="utf-8", errors="ignore") as f:
            return json.load(f)
    except Exception:
        return {}


def calculate_file_hash(filepath: Path) -> str:
    """Calcula hash SHA-256 inmutable de un archivo físico."""
    sha = hashlib.sha256()
    try:
        with open(filepath, "rb") as f:
            while chunk := f.read(65536):
                sha.update(chunk)
        return sha.hexdigest()
    except Exception:
        return ""


def resolve_target_date(date_arg: str, state_dir: Path, fallback_func) -> str:
    """Resuelve la fecha objetivo unificada."""
    if date_arg == "latest":
        sync_file = state_dir / "last_audited_date.txt"
        if sync_file.exists():
            try:
                return sync_file.read_text(encoding="utf-8").strip()
            except Exception:
                pass
        return fallback_func()
    return date_arg


def create_canonical_audit_certificate(
    symbol: str,
    fci_score: float,
    data_integrity: str,
    capture_status: str,
    anomaly_list: Optional[List[Dict]] = None,
    metadata_extra: Optional[Dict] = None
) -> Dict[str, Any]:
    """
    Construye el CERTIFICADO CANÓNICO por símbolo según SYSTEM_CANONICAL.
    Reemplaza definitivamente a create_audit_result_template().
    """
    # Validaciones estrictas contra la Ontología
    if not validate_canonical_enum(data_integrity, DataIntegrity):
        raise ValueError(f"data_integrity inválido: {data_integrity}")
    if not validate_canonical_enum(capture_status, SymbolCaptureStatus):
        raise ValueError(f"symbol_capture_status inválido: {capture_status}")

    envelope = MetadataEnvelope(
        schema_version=SCHEMA_VERSION_AUDIT,
        contract_type=ContractType.AUDIT_CERTIFICATE,
        producer_identity=PRODUCER_AUDIT,
        timestamp_emission=datetime.now(timezone.utc).isoformat()
    )

    certificate = {
        **envelope.to_dict(),
        "symbol": symbol.upper(),
        "fci_score": round(float(fci_score), 4),
        "data_integrity": data_integrity,
        "symbol_capture_status": capture_status,
        "data_anomaly": anomaly_list or [],
        "metadata": metadata_extra or {}
    }
    return certificate


def validate_audit_certificate(cert: Dict[str, Any]) -> bool:
    """Valida que un certificado cumpla con el esquema canónico V3."""
    required = {
        "schema_version", "contract_type", "producer_identity",
        "timestamp_emission", "symbol", "fci_score",
        "data_integrity", "symbol_capture_status"
    }
    if not required.issubset(cert.keys()):
        return False
    if cert.get("contract_type") != ContractType.AUDIT_CERTIFICATE.value:
        return False
    if not validate_canonical_enum(cert.get("data_integrity", ""), DataIntegrity):
        return False
    if not validate_canonical_enum(cert.get("symbol_capture_status", ""), SymbolCaptureStatus):
        return False
    return True


# Funciones legacy para compatibilidad con módulos existentes
def create_audit_result_template(symbol: str, stage: str = "audit") -> Dict[str, Any]:
    """
    Template legacy para compatibilidad con módulos que aún no migran a certificados canónicos.
    Se recomienda migrar a create_canonical_audit_certificate().
    """
    return {
        "symbol": symbol.upper(),
        "stage": stage,
        "status": "PASS",
        "confidence": 100.0,
        "critical_errors": [],
        "warnings": [],
        "metadata": {},
        "timestamp": datetime.now(timezone.utc).isoformat()
    }


def validate_audit_result(result: Dict[str, Any]) -> bool:
    """Valida un resultado de auditoría legacy."""
    required = {"symbol", "stage", "status", "confidence"}
    if not required.issubset(result.keys()):
        return False
    if result.get("confidence", 0) < 0 or result.get("confidence", 0) > 100:
        return False
    return True
