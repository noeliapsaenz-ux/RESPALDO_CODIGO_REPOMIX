import pandas as pd

# Inyección de ruta raíz

    sys.path.insert(0, str(ROOT_DIR))

# Importaciones oficiales del sistema
from SYSTEM_SERVICES.system_paths import AUDIT_REPORTS, AUDIT_STATE, AUDIT_LINEAGE

from SYSTEM_CANONICAL.system_sessions import get_fecha_broker_str

logger = get_system_logger("historical_persistence", "02_AUDIT")

def generate_sha256(raw_str: str) -> str:
    """Genera una firma determinista para asegurar la idempotencia del registro."""
    return hashlib.sha256(raw_str.encode('utf-8')).hexdigest()

def extract_events_for_symbol(symbol: str, fecha: str, capture_item: Optional[Dict],
                             lineage_item: Optional[Dict], consolidated_item: Optional[Dict]) -> List[Dict[str, Any]]:
    """Extrae eventos específicos del feed dignos de ser catalogados en el historial."""
    events = []
    timestamp_str = datetime.now(timezone.utc).isoformat()

    def _create_event(event_name: str, value: Any = 1) -> Dict[str, Any]:
        raw_key = f"{fecha}|{symbol.upper()}|{event_name}|{value}"
        return {
            "event_id": generate_sha256(raw_key),
            "fecha": fecha,
            "timestamp": timestamp_str,
            "symbol": symbol.upper(),
            "event": event_name,
            "value": value
        }

    if capture_item:
        metrics = capture_item.get("integrity_metrics", {})
        missing_files = metrics.get("missing_files_estimated", 0)
        if missing_files > 0:
            events.append(_create_event("missing_files", missing_files))

        neg_gaps = metrics.get("negative_timestamp_gaps", 0)
        if neg_gaps > 0:
            events.append(_create_event("negative_timestamps", neg_gaps))

        stale_pct = metrics.get("stale_ticks_pct", 0.0)
        if stale_pct > STALE_TICKS_EVENT_THRESHOLD:
            events.append(_create_event("stale_ticks_burst", int(stale_pct)))

    if lineage_item:
        stages_meta = lineage_item.get("metadata", {}).get("stages", {})
        if not stages_meta.get("gold", {}).get("exists", False):
            events.append(_create_event("missing_gold_file"))

    if consolidated_item:
        for err in consolidated_item.get("critical_errors", []):
            events.append(_create_event(f"critical_error_{err.get('code', 'generic')}"))

    return events

def find_capture_report(fecha: str) -> Path:
    """Intenta ubicar el reporte de captura del colector en las rutas físicas."""
    posibles_rutas = [
        AUDIT_REPORTS / f"capture_audit_report_{fecha}.json",
        AUDIT_STATE / f"capture_audit_report_{fecha}.json"
    ]
    for r in posibles_rutas:
        if r.exists():
            return r
    return posibles_rutas[0]

def update_history(fecha: str):
    """Consolida todos los insumos de auditoría del día y los vuelca de forma estructurada."""
    capture_report_path = find_capture_report(fecha)
    profile_report_path = AUDIT_REPORTS / f"market_profile_report_{fecha}.json"
    lineage_report_path = AUDIT_LINEAGE / f"lineage_report_{fecha}.json"
    consolidated_report_path = AUDIT_REPORTS / f"audit_report_{fecha}.json"

    capture_report = load_contract_safe(capture_report_path)
    profile_report = load_contract_safe(profile_report_path)
    lineage_report = load_contract_safe(lineage_report_path)
    consolidated_report = load_contract_safe(consolidated_report_path)

    capture_map = {item.get("symbol", "").upper(): item for item in capture_report.get("integrity_summary", [])} if capture_report else {}
    profile_map = {item.get("metadata", {}).get("symbol", "").upper(): item for item in profile_report.get("perfiles", [])} if profile_report else {}
    lineage_map = {sym.upper(): item for sym, item in lineage_report.get("resultados", {}).items()} if lineage_report else {}
    consolidated_map = {item.get("symbol", "").upper(): item for item in consolidated_report.get("resultados_auditoria", [])} if consolidated_report else {}

    history_rows, events_rows = [], []
    symbols = get_symbols()

    for symbol in symbols:
        sym_upper = symbol.upper()
        c_item = capture_map.get(sym_upper, {})
        c_metrics = c_item.get("integrity_metrics", {})
        p_item = profile_map.get(sym_upper, {})
        l_item = lineage_map.get(sym_upper, {})
        con_item = consolidated_map.get(sym_upper, {})

        ticks_total = int(c_item.get("ticks", 0)) if c_item else int(p_item.get("metadata", {}).get("total_ticks", 0))
        spread_mean = float(p_item.get("spread_profile", {}).get("mean_pips", 0.0))
        spread_p95 = float(p_item.get("spread_profile", {}).get("p95_pips", 0.0))
        market_regime = p_item.get("market_regime", {}).get("state", "UNKNOWN")
        trend_strength = float(p_item.get("descriptive_indicators", {}).get("trend", {}).get("trend_strength_pct", 0.0))

        gold_ticks = int(l_item.get("metadata", {}).get("stages", {}).get("gold", {}).get("total_ticks", 0)) if l_item else 0
        parquet_ticks = int(l_item.get("metadata", {}).get("stages", {}).get("parquet", {}).get("total_ticks", 0)) if l_item else 0
        gold_pct = round((gold_ticks / parquet_ticks) * 100.0, 2) if parquet_ticks > 0 else 0.0

        record_id = generate_sha256(f"{fecha}|{sym_upper}")
        history_rows.append({
            "record_id": record_id,
            "fecha": fecha,
            "symbol": sym_upper,
            "capture_conf": float(c_metrics.get("feed_confidence_index", 0.0)),
            "market_conf": float(con_item.get("metadata", {}).get("confianza_perfil", 0.0)) if con_item else 0.0,
            "lineage_conf": float(l_item.get("confidence", 0.0)) if l_item else 0.0,
            "overall_conf": float(con_item.get("confidence", 0.0)) if con_item else 0.0,
            "status": con_item.get("status", "FAIL") if con_item else "FAIL",
            "ticks_total": ticks_total,
            "spread_mean": spread_mean,
            "spread_p95": spread_p95,
            "market_regime": market_regime,
            "trend_strength": trend_strength,
            "gold_percentage": gold_pct
        })
        events_rows.extend(extract_events_for_symbol(sym_upper, fecha, c_item, l_item, con_item))

    df_new = pd.DataFrame(history_rows)
    history_file_path = AUDIT_REPORTS / "confidence_history.parquet"

    if history_file_path.exists():
        try:
            df_old = pd.read_parquet(history_file_path)
            df_old = df_old[~df_old["record_id"].isin(df_new["record_id"])]
            df_combined = pd.concat([df_old, df_new], ignore_index=True)
        except Exception:
            df_combined = df_new
    else:
        df_combined = df_new

    df_combined = df_combined.sort_values(by=["symbol", "fecha"]).reset_index(drop=True)
    df_combined.to_parquet(history_file_path, index=False)

    events_file_path = AUDIT_REPORTS / "audit_events.jsonl"
    existing_event_ids = set()
    if events_file_path.exists():
        try:
            with open(events_file_path, "r", encoding="utf-8") as ef:
                for line in ef:
                    if line.strip():
                        existing_event_ids.add(json.loads(line.strip()).get("event_id"))
        except Exception:
            pass

    try:
        with open(events_file_path, "a", encoding="utf-8") as ef:
            for event in events_rows:
                if event["event_id"] not in existing_event_ids:
                    ef.write(json.dumps(event) + "\n")
    except OSError as e:
        logger.error(f"Error persistiendo eventos en disco: {e}")

def main():
    parser = argparse.ArgumentParser(description="Historical Persistence")
    parser.add_argument("--date", type=str, default="latest")
    args = parser.parse_args()

    fecha = resolve_target_date(args.date, AUDIT_STATE, get_fecha_broker_str)

    logger.info(f"📊 CONSOLIDANDO PERSISTENCIA DE AUDITORÍA HISTÓRICA - FECHA: {fecha}")
    update_history(fecha)
    logger.info(f"✅ Historial y bitácoras para {fecha} actualizados con éxito.")

if __name__ == "__main__":
    main()
