# Umbrales para clasificación
GAP_WARNING_SECONDS = 60
GAP_CRITICAL_SECONDS = 300
BROKER_TZ = get_broker_timezone()

class TickAuditor:
    """Audita un archivo JSONL extrayendo primer y último tick."""

    def __init__(self, filepath: Path):
        self.filepath = filepath
        self.first_seq = None
        self.last_seq = None
        self.first_tm = None
        self.last_tm = None
        self.first_recv_ts = None
        self.last_recv_ts = None
        self.total_ticks = 0
        self.is_valid = False
        self.error = None

    def analyze(self) -> bool:
        """Extrae primer y último tick del archivo."""
        if not self.filepath.exists():
            self.error = "Archivo no encontrado"
            return False

        try:
            with open(self.filepath, 'r', encoding='utf-8') as f:
                first_line = None
                last_line = None
                line_count = 0

                for line in f:
                    line = line.strip()
                    if not line:
                        continue

                    line_count += 1
                    try:
                        tick = json.loads(line)
                        if first_line is None:
                            first_line = tick
                            self.first_seq = tick.get('seq')
                            self.first_tm = tick.get('tm')
                            self.first_recv_ts = tick.get('recv_ts')

                        last_line = tick
                        self.last_seq = tick.get('seq')
                        self.last_tm = tick.get('tm')
                        self.last_recv_ts = tick.get('recv_ts')

                    except json.JSONDecodeError:
                        continue

                self.total_ticks = line_count
                self.is_valid = self.first_seq is not None and self.last_seq is not None
                return self.is_valid

        except Exception as e:
            self.error = str(e)
            return False


class SymbolRecoveryAuditor:
    """Audita la recuperación de un símbolo."""

    def __init__(self, symbol: str):
        self.symbol = symbol.upper()
        self.base_dir = SSD_EVIDENCE_DIR / self.symbol
        self.files = []
        self.audited_files = []
        self.state = None
        self.state_seq = None

    def load_state(self) -> bool:
        """Carga state.json del símbolo."""
        state_file = AUDIT_STATE / f"{self.symbol}_state.json"
        if state_file.exists():
            try:
                self.state = load_contract_safe(state_file)
                self.state_seq = self.state.get('last_sequence')
                return True
            except Exception:
                pass
        return False

    def scan_files(self) -> List[Path]:
        """Escanea archivos JSONL en las subcarpetas."""
        if not self.base_dir.exists():
            return []

        jsonl_files = []
        for date_dir in self.base_dir.iterdir():
            if date_dir.is_dir():
                jsonl_files.extend(date_dir.glob("*.jsonl"))

        jsonl_files.sort(key=lambda f: f.stat().st_mtime)
        self.files = jsonl_files
        return jsonl_files

    def analyze_files(self, max_files: int = 10) -> List[TickAuditor]:
        """Analiza los últimos N archivos."""
        if not self.files:
            return []

        target_files = self.files[-max_files:]
        self.audited_files = []

        for filepath in target_files:
            auditor = TickAuditor(filepath)
            auditor.analyze()
            self.audited_files.append(auditor)

        return self.audited_files

    def audit_recovery(self) -> Dict[str, Any]:
        """Audita la recuperación comparando archivos consecutivos."""
        if len(self.audited_files) < 2:
            return {
                "symbol": self.symbol,
                "overall_status": "INSUFFICIENT_DATA",
                "message": "Menos de 2 archivos para auditar"
            }

        sorted_files = sorted(self.audited_files, key=lambda f: f.first_tm or 0)
        valid_files = [f for f in sorted_files if f.is_valid]

        if len(valid_files) < 2:
            return {
                "symbol": self.symbol,
                "overall_status": "INSUFFICIENT_DATA",
                "message": "Menos de 2 archivos válidos"
            }

        recovery_results = []
        duplicates_found = 0
        regressions_found = 0
        seq_jumps_found = 0
        reset_from_zero_found = False
        total_gap_seconds = 0
        max_gap_seconds = 0
        capture_gaps = []

        prev_file = None
        for curr_file in valid_files:
            if prev_file is None:
                prev_file = curr_file
                continue

            seq_diff = curr_file.first_seq - prev_file.last_seq
            tm_diff_ms = curr_file.first_tm - prev_file.last_tm

            has_duplicate = seq_diff < 1
            has_regression = seq_diff < 0
            has_seq_jump = seq_diff > 1
            has_reset_from_zero = (prev_file.last_seq > 0 and curr_file.first_seq <= 1)

            gap_seconds = tm_diff_ms / 1000 if tm_diff_ms > 0 else 0

            if has_duplicate:
                duplicates_found += 1
            if has_regression:
                regressions_found += 1
            if has_seq_jump:
                seq_jumps_found += 1
            if has_reset_from_zero:
                reset_from_zero_found = True

            total_gap_seconds += gap_seconds if gap_seconds > 0 else 0
            max_gap_seconds = max(max_gap_seconds, gap_seconds)

            if has_reset_from_zero:
                recovery_status = "CRITICAL_RESET"
            elif has_regression:
                recovery_status = "SEQ_REGRESSION"
            elif has_duplicate:
                recovery_status = "SEQ_DUPLICATE"
            elif has_seq_jump:
                recovery_status = "SEQ_JUMP"
            elif seq_diff == 1:
                if gap_seconds > GAP_CRITICAL_SECONDS:
                    recovery_status = "SEQ_OK_WITH_LARGE_GAP"
                elif gap_seconds > GAP_WARNING_SECONDS:
                    recovery_status = "SEQ_OK_WITH_GAP"
                else:
                    recovery_status = "SEQ_CONTINUOUS"
            else:
                recovery_status = "UNKNOWN"

            if gap_seconds > GAP_WARNING_SECONDS:
                capture_gaps.append({
                    "from_file": prev_file.filepath.name,
                    "to_file": curr_file.filepath.name,
                    "gap_seconds": round(gap_seconds, 2),
                    "from_tm": prev_file.last_tm,
                    "to_tm": curr_file.first_tm
                })

            recovery_results.append({
                "from_file": prev_file.filepath.name,
                "to_file": curr_file.filepath.name,
                "from_seq": prev_file.last_seq,
                "to_seq": curr_file.first_seq,
                "seq_diff": seq_diff,
                "tm_gap_seconds": round(gap_seconds, 2),
                "recovery_status": recovery_status
            })

            prev_file = curr_file

        if reset_from_zero_found:
            overall_status = "CRITICAL_RESET_DETECTED"
        elif duplicates_found > 0 or regressions_found > 0:
            overall_status = "SEQ_ANOMALY_DETECTED"
        elif seq_jumps_found > 0:
            overall_status = "SEQ_JUMP_DETECTED"
        elif max_gap_seconds > GAP_CRITICAL_SECONDS:
            overall_status = "RECOVERY_WITH_LARGE_GAP"
        elif max_gap_seconds > GAP_WARNING_SECONDS:
            overall_status = "RECOVERY_WITH_GAP"
        else:
            overall_status = "PERFECT_RECOVERY"

        state_match = False
        state_reset_detected = False
        if self.state_seq is not None and valid_files:
            last_seq = valid_files[-1].last_seq
            state_match = self.state_seq == last_seq

            first_file_seq = valid_files[0].first_seq
            if self.state_seq < first_file_seq:
                state_reset_detected = True

        return {
            "symbol": self.symbol,
            "overall_status": overall_status,
            "state_seq": self.state_seq,
            "first_file_seq": valid_files[0].first_seq if valid_files else None,
            "last_file_seq": valid_files[-1].last_seq if valid_files else None,
            "state_match": state_match,
            "state_reset_detected": state_reset_detected,
            "recovery_checks": recovery_results,
            "duplicates_found": duplicates_found,
            "regressions_found": regressions_found,
            "seq_jumps_found": seq_jumps_found,
            "reset_from_zero_found": reset_from_zero_found,
            "total_gap_seconds": round(total_gap_seconds, 2),
            "max_gap_seconds": round(max_gap_seconds, 2),
            "capture_gaps": capture_gaps,
            "files_analyzed": len(valid_files),
            "timestamp": datetime.now(timezone.utc).isoformat()
        }


def audit_symbol(symbol: str, max_files: int = 10) -> Dict[str, Any]:
    """Audita un símbolo completo."""
    logger.info(f"Auditando {symbol}...")

    auditor = SymbolRecoveryAuditor(symbol)
    auditor.load_state()

    files = auditor.scan_files()
    if not files:
        return {
            "symbol": symbol,
            "overall_status": "NO_DATA",
            "message": "No se encontraron archivos"
        }

    auditor.analyze_files(max_files=max_files)
    result = auditor.audit_recovery()

    logger.info(f"  {symbol}: {result.get('overall_status')} | "
                f"Gap max: {result.get('max_gap_seconds', 0):.1f}s | "
                f"State match: {result.get('state_match', False)}")

    return result


def main():
    parser = argparse.ArgumentParser(description="Recovery Forensics Auditor")
    parser.add_argument("--date", type=str, default="latest",
                       help="Fecha en formato YYYYMMDD")
    parser.add_argument("--max-files", type=int, default=10,
                       help="Número máximo de archivos a analizar por símbolo")
    parser.add_argument("--symbol", type=str, default=None,
                       help="Auditar solo un símbolo específico")
    args = parser.parse_args()

    if args.date == "latest":
        fecha = get_fecha_broker_str()
    else:
        fecha = args.date

    symbols = [args.symbol.upper()] if args.symbol else get_symbols()

    logger.info("=" * 70)
    logger.info(f"🔍 RECOVERY FORENSICS - Fecha: {fecha}")
    logger.info(f"📊 Símbolos: {len(symbols)}")
    logger.info("=" * 70)

    results = {}
    for symbol in symbols:
        results[symbol] = audit_symbol(symbol, args.max_files)

    AUDIT_RECOVERY.mkdir(parents=True, exist_ok=True)

    for symbol, result in results.items():
        if result.get("overall_status") != "NO_DATA":
            output_file = AUDIT_RECOVERY / f"recovery_{symbol}_{fecha}.json"
            save_contract_atomic(result, output_file)
            logger.info(f"💾 Guardado: {output_file.name}")

    consolidated = {
        "fecha": fecha,
        "timestamp": datetime.now(timezone.utc).isoformat(),
        "symbols": results,
        "summary": {
            "total": len(results),
            "perfect": sum(1 for r in results.values() if r.get("overall_status") == "PERFECT_RECOVERY"),
            "with_gap": sum(1 for r in results.values() if r.get("overall_status") in ["RECOVERY_WITH_GAP", "RECOVERY_WITH_LARGE_GAP"]),
            "critical": sum(1 for r in results.values() if r.get("overall_status") in ["CRITICAL_RESET_DETECTED", "SEQ_ANOMALY_DETECTED", "SEQ_JUMP_DETECTED"]),
            "state_consistent": sum(1 for r in results.values() if r.get("state_match", False))
        }
    }

    consolidated_file = AUDIT_RECOVERY / f"recovery_consolidated_{fecha}.json"
    save_contract_atomic(consolidated, consolidated_file)

    logger.info("=" * 70)
    logger.info(f"✅ RECOVERY FORENSICS COMPLETADO")
    logger.info(f"   Perfectos: {consolidated['summary']['perfect']}")
    logger.info(f"   Con gaps: {consolidated['summary']['with_gap']}")
    logger.info(f"   Críticos: {consolidated['summary']['critical']}")
    logger.info(f"   State OK: {consolidated['summary']['state_consistent']}")
    logger.info("=" * 70)


if __name__ == "__main__":
    try:
        main()
    except KeyboardInterrupt:
        print("\n❌ Interrumpido")
        sys.exit(1)
    except Exception as e:
        logger.error(f"Error crítico: {e}", exc_info=True)
        sys.exit(1)
