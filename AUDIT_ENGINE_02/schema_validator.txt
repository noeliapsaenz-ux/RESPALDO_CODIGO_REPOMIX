import sys
import argparse
from pathlib import Path
from datetime import datetime, timezone
from typing import Dict, Any
import pandas as pd

# Inyección de ruta raíz
ROOT_DIR = Path(__file__).resolve().parents[1]
if str(ROOT_DIR) not in sys.path:
    sys.path.insert(0, str(ROOT_DIR))

# Importaciones oficiales del sistema
from SYSTEM_SERVICES.system_paths import AUDIT_STATE, AUDIT_MAESTROS, AUDIT_GOLD
from SYSTEM_CANONICAL.system_symbols import get_symbols
from SYSTEM_CANONICAL.system_sessions import get_fecha_broker_str
from SYSTEM_SERVICES.system_logger import get_system_logger
from SYSTEM_CONTRACTS.contract_manager import PARQUET_SCHEMAS, AUDIT_VERSION

from AUDIT_ENGINE_02.audit_contracts import resolve_target_date

logger = get_system_logger("schema_validator", "02_AUDIT")

COLUMNAS_REQUERIDAS_MAESTRO = PARQUET_SCHEMAS["parquet_required_columns"]

def verificar_parquet_completo(symbol: str, fecha: str) -> Dict[str, Any]:
    """Verifica físicamente la disponibilidad de archivos parquet y su estructura."""
    resultado = {
        'symbol': symbol,
        'maestro_existe': False,
        'maestro_filas': 0,
        'gold_existe': False,
        'gold_filas': 0,
        'gold_percent': 0.0,
        'estructura_ok': False,
        'columnas_faltantes': [],
        'avg_quality_gold': "N/A",
        'errores': []
    }

    maestro_path = AUDIT_MAESTROS / f"ticks_maestro_{symbol}_{fecha}.parquet"
    if not maestro_path.exists():
        resultado['errores'].append("Archivo de datos Parquet Maestro ausente.")
        return resultado

    resultado['maestro_existe'] = True
    try:
        df_maestro = pd.read_parquet(maestro_path)
        resultado['maestro_filas'] = len(df_maestro)
        resultado['columnas_faltantes'] = [col for col in COLUMNAS_REQUERIDAS_MAESTRO if col not in df_maestro.columns]
        resultado['estructura_ok'] = len(resultado['columnas_faltantes']) == 0
    except Exception as e:
        resultado['errores'].append(f"Fallo de lectura de Parquet Maestro: {e}")
        return resultado

    gold_path = AUDIT_GOLD / f"{symbol}_{fecha}_gold.parquet"
    if gold_path.exists():
        resultado['gold_existe'] = True
        try:
            df_gold = pd.read_parquet(gold_path)
            resultado['gold_filas'] = len(df_gold)
            if resultado['maestro_filas'] > 0:
                resultado['gold_percent'] = round((resultado['gold_filas'] / resultado['maestro_filas']) * 100, 1)

            if 'quality_score' in df_gold.columns:
                resultado['avg_quality_gold'] = round(float(df_gold['quality_score'].mean()), 2)
        except Exception as e:
            resultado['errores'].append(f"Fallo de lectura de Parquet Gold: {e}")

    return resultado

def main():
    parser = argparse.ArgumentParser(description="Schema Validator")
    parser.add_argument("--date", type=str, default="latest")
    args = parser.parse_args()

    fecha = resolve_target_date(args.date, AUDIT_STATE, get_fecha_broker_str)
    logger.info(f"📋 VERIFICANDO INTEGRIAD DE COLUMNAS DE ARCHIVOS - FECHA: {fecha} (V. {AUDIT_VERSION})")

    correctos = 0
    symbols = get_symbols()
    for symbol in symbols:
        res = verificar_parquet_completo(symbol, fecha)
        if res['maestro_existe'] and res['estructura_ok']:
            correctos += 1
            logger.info(f"  • {symbol:8} | Maestro: {res['maestro_filas']:9,} | Gold: {res['gold_filas']:9,} ({res['gold_percent']}%) | Columnas: OK | Calidad Gold: {res['avg_quality_gold']}")
        else:
            logger.warning(f"  ❌ {symbol:8} | Errores de verificación de estructura: {res['errores']}")

    logger.info(f"\n📊 Esquemas correctos evaluados: {correctos}/{len(symbols)} archivos.")

if __name__ == "__main__":
    main()
