def cargar_historial_volatilidad(history_file: Path) -> Dict[str, float]:
    default_vol = {sym: 0.0015 for sym in get_symbols()}
    if not history_file.exists(): return default_vol
    try:
        df = pd.read_parquet(history_file)
        if df.empty: return default_vol
        stats = df.groupby("symbol")["realized_vol_mean"].mean().to_dict()
        return {sym.upper(): float(val) for sym, val in stats.items()}
    except Exception as e:
        logger.debug(f"Error cargando base histórica de volatilidad: {e}")
        return default_vol

def sincronizar_e_intersecar_estados(fecha: str) -> Tuple[pd.DataFrame, Dict[str, pd.DataFrame]]:
    series_map = {}
    closes_map = {}
    for symbol in get_symbols():
        symbol = symbol.upper()
        state_file = QUANT_BARS / symbol / f"{symbol}_{fecha}_market_state.parquet"
        if state_file.exists():
            try:
                df = pd.read_parquet(state_file)
                if not df.empty:
                    df_cleaned = df.drop_duplicates(subset=["timestamp"], keep="first").copy()
                    series_map[symbol] = df_cleaned.set_index("timestamp").sort_index()
                    closes_map[symbol] = series_map[symbol]["close"]
            except Exception as e:
                logger.error(f"❌ {symbol}: Error de alineación: {e}")
    if not series_map: return pd.DataFrame(), {}
    df_closes = pd.DataFrame(closes_map).ffill().bfill()
    df_returns = np.log(df_closes / df_closes.shift(1)).fillna(0.0)
    return df_returns, series_map

def calcular_volatilidad_sistémica(series_map: Dict[str, pd.DataFrame], vol_hist: Dict[str, float]) -> Dict[str, Any]:
    t = SYSTEMIC_THRESHOLDS
    vol_ratios = [float(df["realized_volatility"].iloc[-1]) / (vol_hist.get(sym, 0.0015) + 1e-9) 
                  for sym, df in series_map.items() if "realized_volatility" in df.columns]
    if not vol_ratios: return {"score": 50.0, "regime": "NORMAL", "confidence": 0.0}
    svi = float(np.mean(vol_ratios)) * 100.0
    regime = "QUIET" if svi < t["volatility_svi_quiet"] else ("NORMAL" if svi < t["volatility_svi_normal"] else ("ACTIVE" if svi < t["volatility_svi_active"] else "EXPLOSIVE"))
    return {"score": round(svi, 1), "regime": regime, "confidence": 100.0}

def calcular_liquidez_sistémica(series_map: Dict[str, pd.DataFrame]) -> Dict[str, Any]:
    t = SYSTEMIC_THRESHOLDS
    spread_ratios = [float(df["spread_mean"].iloc[-1]) / (get_spread_limit(sym) + 1e-9) 
                     for sym, df in series_map.items() if "spread_mean" in df.columns]
    if not spread_ratios: return {"score": 50.0, "regime": "NORMAL", "confidence": 0.0}
    gli = float(np.clip(100.0 - (np.mean(spread_ratios) - 0.5) * 100.0, 0.0, 100.0))
    regime = "OPTIMAL" if gli >= t["liquidity_gli_optimal"] else ("NORMAL" if gli >= t["liquidity_gli_normal"] else "CRITICAL")
    return {"score": round(gli, 1), "regime": regime, "confidence": 100.0}

def calcular_cohesion_sistémica(df_returns: pd.DataFrame) -> Dict[str, Any]:
    t = SYSTEMIC_THRESHOLDS
    if len(df_returns) < 15: return {"score": 50.0, "regime": "UNKNOWN", "confidence": 0.0}
    corr_hybrid = (df_returns.corr(method="pearson") + df_returns.corr(method="spearman")) / 2.0
    sc_mean = np.nanmean(np.abs(corr_hybrid.values))
    regime = "IDIOSYNCRATIC" if sc_mean < t["cohesion_idiosyncratic"] else ("ROTATION" if sc_mean < t["cohesion_rotation"] else "COHESIVE / SYSTEMIC")
    return {"score": round(sc_mean * 100.0, 1), "mean_absolute_correlation": round(sc_mean, 4), "regime": regime, "confidence": 100.0}

def evaluar_sentimiento_riesgo(df_returns: pd.DataFrame) -> Dict[str, Any]:
    t = SYSTEMIC_THRESHOLDS
    risk_vals = [df_returns[s].sum() for s in get_symbols() if get_symbol_specification(s).is_risk_asset and s in df_returns]
    safe_vals = [df_returns[s].sum() for s in get_symbols() if get_symbol_specification(s).is_safe_haven and s in df_returns]
    if not risk_vals or not safe_vals: return {"score": 0.0, "regime": "NEUTRAL", "confidence": 0.0}
    grai = (float(np.mean(risk_vals)) - float(np.mean(safe_vals))) * 100.0
    regime = "RISK_ON" if grai > t["risk_appetite_on"] else ("RISK_OFF" if grai < t["risk_appetite_off"] else "NEUTRAL")
    return {"score": round(grai, 3), "regime": regime, "confidence": 100.0}

def evaluar_actividad_global(series_map: Dict[str, pd.DataFrame]) -> Dict[str, Any]:
    t = SYSTEMIC_THRESHOLDS
    total_tps = sum(float(df["tps_abs"].iloc[-1]) for df in series_map.values() if "tps_abs" in df.columns)
    regime = "QUIET" if total_tps < t["activity_quiet_tps"] else ("NORMAL" if total_tps < t["activity_normal_tps"] else "ACTIVE")
    return {"cumulative_tps": round(total_tps, 2), "regime": regime, "confidence": 100.0}

def generar_contrato_estado_global(fecha: str) -> bool:
    history_file = QUANT_HISTORY / "daily_history_30d.parquet"
    output_file = QUANT_OUTPUTS / f"global_regime_{fecha}.json"
    df_returns, series_map = sincronizar_e_intersecar_estados(fecha)
    if not series_map: return False
    
    vol_hist = cargar_historial_volatilidad(history_file)
    vol_sys = calcular_volatilidad_sistémica(series_map, vol_hist)
    liq_sys = calcular_liquidez_sistémica(series_map)
    coh_sys = calcular_cohesion_sistémica(df_returns)
    risk_sys = evaluar_sentimiento_riesgo(df_returns)
    act_sys = evaluar_actividad_global(series_map)

    # Permitir ejecución basada en estados
    allow_scalping = (vol_sys["regime"] != "EXPLOSIVE" and liq_sys["regime"] != "CRITICAL")
    allow_swing = (vol_sys["regime"] != "QUIET" and coh_sys["regime"] != "COHESIVE / SYSTEMIC")
    
    payload = {
        "fecha": fecha, "timestamp_generacion": datetime.now(timezone.utc).isoformat(),
        "global_market_regime": {"vol": vol_sys, "liq": liq_sys, "coh": coh_sys, "risk": risk_sys, "act": act_sys},
        "global_operational_permits": {"allow_scalping": allow_scalping, "allow_swing": allow_swing, "allow_portfolio": (liq_sys["regime"] != "CRITICAL")}
    }
    
    try:
        save_contract_atomic(payload, output_file)
        # (Continuar con la emisión del Quant Certificate...)
        return True
    except Exception as e:
        logger.error(f"Error: {e}")
        return False
