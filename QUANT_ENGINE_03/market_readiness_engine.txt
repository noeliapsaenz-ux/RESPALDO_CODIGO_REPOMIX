def evaluar_readiness_simbolo(symbol: str, fecha: str, audit_status: str, ref_vol: float) -> dict[str, any]:
    symbol = symbol.upper()
    state_file = QUANT_BARS / symbol / f"{symbol}_{fecha}_market_state.parquet"
    default_metrics = {"spread_mean": 99.0, "jitter_mean_ms": 99.0, "tps_mean": 0.0, "quality_mean": 0.0}
    
    if audit_status != SymbolCaptureStatus.CONFIABLE.value:
        logger.warning(f"❌ {symbol}: VETO DURO H-03 aplicado. No confiable para operar.")
        return {
            "symbol": symbol, "market_readiness_status": MarketReadinessStatus.VETOED.value,
            "readiness_by_style": {
                "scalping": {"status": "FAIL", "readiness_score": 0.0, "reason": "Capture Veto"},
                "intraday": {"status": "FAIL", "readiness_score": 0.0, "reason": "Capture Veto"},
                "swing": {"status": "FAIL", "readiness_score": 0.0, "reason": "Capture Veto"}
            }, "metadata": {"evaluated_metrics": default_metrics, "fci_captura": 0.0}
        }

    if not state_file.exists():
        logger.warning(f"❌ {symbol}: No se encuentra el parquet de estado para {fecha}.")
        return {
            "symbol": symbol, "market_readiness_status": MarketReadinessStatus.INSUFFICIENT_DATA.value,
            "readiness_by_style": {
                "scalping": {"status": "FAIL", "readiness_score": 0.0, "reason": "No State Data"},
                "intraday": {"status": "FAIL", "readiness_score": 0.0, "reason": "No State Data"},
                "swing": {"status": "FAIL", "readiness_score": 0.0, "reason": "No State Data"}
            }, "metadata": {"evaluated_metrics": default_metrics, "fci_captura": 100.0}
        }

    try:
        df = pd.read_parquet(state_file)
        if df.empty: raise ValueError("Empty state parquet")
    except Exception as e:
        logger.error(f"❌ {symbol}: Error leyendo el snapshot de estado: {e}")
        return {
            "symbol": symbol, "market_readiness_status": MarketReadinessStatus.INSUFFICIENT_DATA.value,
            "readiness_by_style": {
                "scalping": {"status": "FAIL", "readiness_score": 0.0, "reason": f"Read Error: {e}"},
                "intraday": {"status": "FAIL", "readiness_score": 0.0, "reason": f"Read Error: {e}"},
                "swing": {"status": "FAIL", "readiness_score": 0.0, "reason": f"Read Error: {e}"}
            }, "metadata": {"evaluated_metrics": default_metrics, "fci_captura": 100.0}
        }

    limit = get_spread_limit(symbol)
    tick_count = int(df["tick_count"].sum())
    spread_mean = float(df["spread_mean"].mean())
    spread_cv = float(df["spread_cv"].mean())
    spread_skew = float(df["spread_skew"].mean())
    tps_abs = float(df["tps_abs"].mean())
    idle_ratio = float(df["idle_ratio"].mean())
    quote_balance = float(df["quote_balance"].mean())
    realized_vol = float(df["realized_volatility"].mean())
    burstiness = float(df["burstiness"].mean())
    quality_mean = float(df["quality_score_mean"].mean())
    jitter = float(df["jitter_ms"].mean())
    
    s_spread = cs.calculate_spread_score(spread_mean, spread_cv, spread_skew, limit, tick_count)["score"]
    s_liq = cs.calculate_liquidity_score(tps_abs, spread_cv, idle_ratio, quote_balance, tick_count)["score"]
    s_vol = cs.calculate_volatility_score(realized_vol, burstiness, ref_vol, tick_count)["score"]
    s_integ = cs.calculate_integrity_score(quality_mean, tick_count)["score"]

    persistence_val = 50.0
    if "regime_duration_bars" in df.columns:
        persistence_val = float(df["regime_duration_bars"].mean())

    evaluated_metrics = {"spread_mean": round(spread_mean, 2), "jitter_mean_ms": round(jitter, 2), "tps_mean": round(tps_abs, 2), "quality_mean": round(quality_mean, 1)}
    readiness_by_style = {}
    any_pass = False

    for style in ["scalping", "intraday", "swing"]:
        weights = READINESS_STYLE_WEIGHTS[style]
        score = 0.0
        if "spread" in weights: score += s_spread * weights["spread"]
        if "execution" in weights: score += s_liq * weights["execution"]
        if "volatility" in weights: score += s_vol * weights["volatility"]
        if "anomaly" in weights: score += s_integ * weights["anomaly"]
        if "persistence" in weights: score += min(100.0, persistence_val * 2.0) * weights["persistence"]
        
        veto_rules = STYLE_VETO_RULES[style]
        fci_min = veto_rules.get("fci_min", 70.0)
        spread_min = veto_rules.get("spread_score_min", 40.0)
        status = "PASS"
        reason = "Aprobado"
        
        if quality_mean < fci_min: status = "FAIL"; reason = f"FCI por debajo de {fci_min}%"
        elif s_spread < spread_min: status = "FAIL"; reason = f"Spread Score por debajo de {spread_min}"
            
        if style == "swing": min_threshold = READINESS_THRESHOLDS["min_score_swing"]
        elif style == "intraday": min_threshold = READINESS_THRESHOLDS["min_score_intraday"]
        else: min_threshold = READINESS_THRESHOLDS["min_score_scalping"]
        
        if status == "PASS" and score < min_threshold: status = "FAIL"; reason = f"Readiness Score {score:.1f} por debajo de {min_threshold}"
        if status == "PASS": any_pass = True

        readiness_by_style[style] = {"status": status, "readiness_score": round(score, 2), "reason": reason}

    market_status = MarketReadinessStatus.OPERABLE.value if any_pass else MarketReadinessStatus.VETOED.value
    return {"symbol": symbol, "market_readiness_status": market_status, "readiness_by_style": readiness_by_style, "metadata": {"evaluated_metrics": evaluated_metrics, "fci_captura": round(quality_mean, 2)}}

def main():
    parser = argparse.ArgumentParser(description="Market Readiness Engine V2.2")
    parser.add_argument("--date", type=str, default="latest")
    args = parser.parse_args()

    fecha = args.date if args.date != "latest" else get_fecha_broker_str()
    logger.info("=" * 80)
    logger.info(f"🧠 INICIANDO EVALUACIÓN DE PREPARACIÓN OPERACIONAL: {fecha}")
    logger.info("=" * 80)

    audit_contract_path = AUDIT_STATE / "audit_execution_contract.json"
    audit_status_map = {}
    
    if audit_contract_path.exists():
        try:
            audit_data = load_contract_safe(audit_contract_path)
            if validate_contract_schema(audit_data, ContractType.AUDIT.value, fecha):
                for sym_key, sym_val in audit_data.get("contracts", {}).items():
                    audit_status_map[sym_key.upper()] = sym_val.get("symbol_capture_status", SymbolCaptureStatus.NO_CONFIABLE.value)
                logger.info(f"✅ Contrato auditado verificado de forma alineada para {fecha}.")
            else:
                logger.critical(f"🛑 Error de validación de contrato de auditoría.")
                for s in get_symbols(): audit_status_map[s.upper()] = SymbolCaptureStatus.NO_CONFIABLE.value
        except Exception as e:
            logger.error(f"❌ Error al evaluar contrato unificado de auditoría: {e}")
            for s in get_symbols(): audit_status_map[s.upper()] = SymbolCaptureStatus.NO_CONFIABLE.value
    else:
        logger.warning("⚠️ No se encontró el contrato de auditoría en disco. Todos los símbolos vetados en cascada.")
        for s in get_symbols(): audit_status_map[s.upper()] = SymbolCaptureStatus.NO_CONFIABLE.value

    history_file = QUANT_HISTORY / "daily_history_30d.parquet"
    # 🟢 CORREGIDO: Uso de la constante canónica en lugar de 0.0015硬codeado
    default_vols = {sym: COLD_START_VOLATILITY for sym in get_symbols()}
    if history_file.exists():
        try:
            df_hist = pd.read_parquet(history_file)
            if not df_hist.empty: default_vols = df_hist.groupby("symbol")["realized_vol_mean"].mean().to_dict()
        except Exception as e: logger.debug(f"Error procesando volatilidad de referencia: {e}")

    readiness_results = []
    for symbol in get_symbols():
        sym_upper = symbol.upper()
        audit_status = audit_status_map.get(sym_upper, SymbolCaptureStatus.NO_CONFIABLE.value)
        ref_vol = default_vols.get(sym_upper, COLD_START_VOLATILITY)
        eval_data = evaluar_readiness_simbolo(sym_upper, fecha, audit_status, ref_vol)
        readiness_results.append(eval_data)
        logger.info(f"   • {sym_upper:6}: Scalping: {eval_data['readiness_by_style']['scalping']['status']} | Intraday: {eval_data['readiness_by_style']['intraday']['status']} | Swing: {eval_data['readiness_by_style']['swing']['status']} --> Estado: {eval_data['market_readiness_status']}")

    readiness_entries = []
    for item in readiness_results:
        scores = {k: v["readiness_score"] for k, v in item["readiness_by_style"].items()}
        readiness_entries.append(create_readiness_entry(item["symbol"], item["market_readiness_status"], scores, item["metadata"]["fci_captura"]))
    
    readiness_cert = create_canonical_readiness_certificate(fecha, readiness_entries)
    readiness_cert["resultados_readiness"] = readiness_results
    readiness_cert["contract_type"] = ContractType.READINESS.value

    output_file = QUANT_READINESS / f"readiness_report_{fecha}.json"
    try:
        save_contract_atomic(readiness_cert, output_file)
        logger.info(f"✅ Certificado de Readiness guardado con éxito en: {output_file.name}")
    except Exception as e:
        logger.error(f"❌ Error guardando el certificado de Readiness: {e}")
        sys.exit(1)

if __name__ == "__main__": 
    main()
