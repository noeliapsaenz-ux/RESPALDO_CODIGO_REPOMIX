import sys
from pathlib import Path
from datetime import datetime, timezone
from typing import Dict, Any, Optional, List

# Inyección dinámica de la raíz del proyecto para resolver dependencias
root_path = Path(__file__).resolve().parents[1]
if str(root_path) not in sys.path:
    sys.path.insert(0, str(root_path))

# Imports directos desde la Capa Canónica Centralizada (Sin quant_config.py)
from SYSTEM_CANONICAL.system_canonical import (
    MetadataEnvelope,
    validate_canonical_enum
)
from SYSTEM_CANONICAL.system_states import (
    MarketRegime,
    MarketActivityState,
    MarketReadinessStatus,
    ContractType
)

# Constantes unificadas de la Fase 3
SCHEMA_VERSION_QUANT = "2.0"
PRODUCER_QUANT = "QUANT_ENGINE_03"


def create_quant_symbol_entry(
    symbol: str,
    market_regime: str,
    market_liquidity_profile: float,
    market_activity_state: str
) -> Dict[str, Any]:
    """Construye la entrada canónica por símbolo para el Quant Certificate."""
    if not validate_canonical_enum(market_regime, MarketRegime):
        raise ValueError(f"market_regime inválido: {market_regime}")
    if not validate_canonical_enum(market_activity_state, MarketActivityState):
        raise ValueError(f"market_activity_state inválido: {market_activity_state}")

    return {
        "symbol": symbol.upper(),
        "market_regime": market_regime,
        "market_liquidity_profile": round(float(market_liquidity_profile), 4),
        "market_activity_state": market_activity_state
    }


def create_canonical_quant_certificate(
    fecha: str,
    symbols_data: List[Dict[str, Any]],
    systemic_metrics: Optional[Dict[str, Any]] = None
) -> Dict[str, Any]:
    """Construye el QUANT_CERTIFICATE consolidado con cabecera de gobernanza."""
    envelope = MetadataEnvelope(
        schema_version=SCHEMA_VERSION_QUANT,
        contract_type=ContractType.PORTFOLIO,  # Utiliza el tipo canónico inmutable
        producer_identity=PRODUCER_QUANT,
        timestamp_emission=datetime.now(timezone.utc).isoformat()
    )

    return {
        **envelope.to_dict(),
        "fecha": fecha,
        "systemic_metrics": systemic_metrics or {},
        "symbols": symbols_data
    }


def create_canonical_readiness_certificate(
    fecha: str,
    readiness_entries: List[Dict[str, Any]]
) -> Dict[str, Any]:
    """Construye el certificado de Readiness con cabecera de gobernanza."""
    envelope = MetadataEnvelope(
        schema_version=SCHEMA_VERSION_QUANT,
        contract_type=ContractType.READINESS,  # Utiliza el tipo de reporte oficializado
        producer_identity=f"{PRODUCER_QUANT}/readiness",
        timestamp_emission=datetime.now(timezone.utc).isoformat()
    )

    return {
        **envelope.to_dict(),
        "fecha": fecha,
        "readiness_by_symbol": readiness_entries
    }


def create_readiness_entry(
    symbol: str,
    market_readiness_status: str,
    readiness_scores: Dict[str, float],
    capture_status: str
) -> Dict[str, Any]:
    """Construye la entrada de readiness por símbolo."""
    if not validate_canonical_enum(market_readiness_status, MarketReadinessStatus):
        raise ValueError(f"market_readiness_status inválido: {market_readiness_status}")

    return {
        "symbol": symbol.upper(),
        "market_readiness_status": market_readiness_status,
        "symbol_capture_status": capture_status,
        "readiness_scores": readiness_scores
    }


def validate_quant_certificate(cert: Dict[str, Any]) -> bool:
    """Valida que un Quant Certificate cumpla la firma canónica."""
    required = {"schema_version", "contract_type", "producer_identity",
                "timestamp_emission", "fecha", "symbols"}
    if not required.issubset(cert.keys()):
        return False
    if cert.get("contract_type") != ContractType.PORTFOLIO.value:
        return False
    return True
