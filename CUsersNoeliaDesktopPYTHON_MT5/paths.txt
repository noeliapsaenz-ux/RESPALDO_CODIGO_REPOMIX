# -*- coding: utf-8 -*-
"""
==========================================================================
PATHS - Mapa de rutas y zonas de almacenamiento
==========================================================================
Ubicación: CARPETA_RAIZ
Dependencias: settings (USAR_RAMDISK)
Responsabilidad: Definir las rutas absolutas de las 7 zonas de almacenamiento:
                 HOT_RAM (RAMDisk/SSD), SSD_EVIDENCE, MASTER_ARCHIVE,
                 AUDIT_WORKSPACE, OPS_INTEL, DECISION_WORKSPACE, STATE y LOGS.
                 Incluye detección automática de disponibilidad de RAMDisk
                 y limpieza de carpetas obsoletas/heredadas.
==========================================================================
"""
import sys
import shutil
from pathlib import Path
from settings import USAR_RAMDISK

# Forzamos la escritura de datos a C:\MT5_DATA
BASE_DIR = Path(r"C:\MT5_DATA")
RAMDISK_PATH = Path("M:\\")

# Verificar pasivamente si el RAMDisk está montado y es escribible
ramdisk_usable = False
if USAR_RAMDISK:
    try:
        if RAMDISK_PATH.exists():
            # Prueba rápida de escritura
            test_file = RAMDISK_PATH / ".ramdisk_test"
            test_file.write_text("ok", encoding="utf-8")
            test_file.unlink(missing_ok=True)
            ramdisk_usable = True
    except Exception:
        ramdisk_usable = False

# ZONA 1: Caliente (Ticks activos en vivo). Fallback automático al SSD si falla la RAM
HOT_RAM_DIR = RAMDISK_PATH / "tick_data" if ramdisk_usable else BASE_DIR / "01_HOT_RAM"

if USAR_RAMDISK and not ramdisk_usable:
    print("⚠️ ADVERTENCIA: RAMDisk (unidad M:\\) no disponible o sin de permisos. Redirigiendo a SSD local.", file=sys.stderr)

# ZONA 2: Evidencia (Buffer local persistente antes del búnker)
SSD_EVIDENCE_DIR = BASE_DIR / "02_SSD_EVIDENCE"

# ZONA 3: Búnker (Almacenamiento comprimido e inmutable)
MASTER_ARCHIVE_DIR = BASE_DIR / "03_MASTER_ARCHIVE"
ARCHIVE_SAFE_DIR = MASTER_ARCHIVE_DIR / "SAFE"
ARCHIVE_QUARANTINE_DIR = MASTER_ARCHIVE_DIR / "QUARANTINE"

# ZONA 4: Auditoría (Mesa de trabajo de calidad)
AUDIT_WORKSPACE_DIR = BASE_DIR / "04_AUDIT_WORKSPACE"
AUDIT_MAESTROS_DIR = AUDIT_WORKSPACE_DIR / "MAESTROS"
AUDIT_GOLD_DIR = AUDIT_WORKSPACE_DIR / "GOLD"
AUDIT_REPORTES_DIR = AUDIT_WORKSPACE_DIR / "REPORTS"
AUDIT_STATE_DIR = AUDIT_WORKSPACE_DIR / "STATE"
AUDIT_FLOW_DIR = AUDIT_WORKSPACE_DIR / "FLOW"
AUDIT_RECOVERY_DIR = AUDIT_WORKSPACE_DIR / "RECOVERY"
AUDIT_LINEAGE_DIR = AUDIT_WORKSPACE_DIR / "LINEAGE"

# ZONA 5: Inteligencia Operacional (Unificada con 05_QUANT_WORKSPACE)
OPS_INTEL_DIR = BASE_DIR / "05_QUANT_WORKSPACE"
QUANT_WORKSPACE_DIR = OPS_INTEL_DIR
OI_OUTPUT_DIR = OPS_INTEL_DIR / "OUTPUTS"
OI_CONFIG_DIR = OPS_INTEL_DIR / "CONFIG"
OI_FEEDBACK_DIR = OPS_INTEL_DIR / "FEEDBACK"
DYNAMIC_SYMBOL_CONFIG_PATH = OI_FEEDBACK_DIR / "dynamic_symbol_config.json"
OI_LOCK_FILE = OPS_INTEL_DIR / "market_readiness.lock"

# NOTA: Se redefine la ruta histórica apuntándola directamente a processed_data/history
OI_HISTORICAL_DIR = OPS_INTEL_DIR / "processed_data" / "history"

# ZONA 6: Motor de Decisiones
DECISION_WORKSPACE_DIR = BASE_DIR / "06_DECISION_WORKSPACE"

# ZONA 7: Estados y Logs
STATE_DIR = BASE_DIR / "STATE"
LOGS_MASTER_DIR = BASE_DIR / "LOGS_MASTER"
LOGS_CRITICAL_DIR = LOGS_MASTER_DIR / "CRITICAL"

# Aliases de compatibilidad requeridos por el pipeline
ACTIVE_DATA_DIR = HOT_RAM_DIR
SSD_BACKUP_DIR = SSD_EVIDENCE_DIR
LOGS_UNIFIED_DIR = LOGS_MASTER_DIR
PROCESSED_DATA_DIR = OPS_INTEL_DIR / "processed_data"

def create_directories():
    """Crea de forma segura la estructura de carpetas de escritura y depura carpetas legacy."""
    # Lista de carpetas deseadas en el nuevo mapa organizado de producción
    dirs_to_create = [
        HOT_RAM_DIR, SSD_EVIDENCE_DIR, MASTER_ARCHIVE_DIR, ARCHIVE_SAFE_DIR, ARCHIVE_QUARANTINE_DIR,
        AUDIT_WORKSPACE_DIR, AUDIT_MAESTROS_DIR, AUDIT_GOLD_DIR, AUDIT_REPORTES_DIR, AUDIT_STATE_DIR,
        AUDIT_FLOW_DIR, AUDIT_RECOVERY_DIR, AUDIT_LINEAGE_DIR,
        OPS_INTEL_DIR, OI_OUTPUT_DIR, OI_CONFIG_DIR, OI_FEEDBACK_DIR,
        QUANT_WORKSPACE_DIR, DECISION_WORKSPACE_DIR, STATE_DIR, LOGS_MASTER_DIR, LOGS_CRITICAL_DIR
    ]
    
    # 1. Crear directorios vigentes
    for d in dirs_to_create:
        try:
            d.mkdir(parents=True, exist_ok=True)
        except Exception as e:
            print(f"Error crítico al estructurar el directorio {d}: {e}", file=sys.stderr)

    # 2. DEPURE AUTOMÁTICO: Buscar y remover la carpeta obsoleta OUTPUTS/HISTORICAL si existe
    legacy_historical = OI_OUTPUT_DIR / "HISTORICAL"
    if legacy_historical.exists() and legacy_historical.is_dir():
        try:
            shutil.rmtree(legacy_historical, ignore_errors=True)
        except Exception:
            pass  # Falla silenciosa si el sistema operativo tiene el directorio bloqueado temporalmente
