class DailyFolderHandler(logging.Handler):
    """Handler dinámico que distribuye logs en carpetas según fecha del Broker."""
    terminator = '\n'

    def __init__(self, base_dir: Path, category: str, filename: str):
        super().__init__()
        self.base_dir = base_dir
        self.category = category  # Ej: "01_CAPTURE", "02_AUDIT", "CRITICAL"
        self.filename = filename
        self.current_date = None
        self.stream = None
        self._lock = threading.Lock()
        self._update_path()

    def _update_path(self):
        """Calcula el directorio diario y abre el puntero de archivo físico."""
        fecha_str = datetime.now(BROKER_TZ).strftime('%Y%m%d')
        
        if fecha_str != self.current_date:
            self.current_date = fecha_str
            
            if self.category == "CRITICAL":
                folder = self.base_dir / "CRITICAL"
            else:
                folder = self.base_dir / fecha_str / self.category
            
            folder.mkdir(parents=True, exist_ok=True)
            self.file_path = folder / self.filename
            
            if self.stream:
                try: 
                    self.stream.close()
                except Exception: 
                    pass
            
            self.stream = open(self.file_path, 'a', encoding='utf-8')

    def emit(self, record):
        """Escribe el registro evaluando si ha ocurrido un rollover diario."""
        try:
            fecha_str = datetime.now(BROKER_TZ).strftime('%Y%m%d')
            if fecha_str != self.current_date:
                with self._lock:
                    if fecha_str != self.current_date:
                        self._update_path()
            
            msg = self.format(record)
            self.stream.write(msg + self.terminator)
            self.flush()
        except Exception:
            self.handleError(record)

    def flush(self):
        self.acquire()
        try:
            if self.stream and not self.stream.closed:
                self.stream.flush()
        except Exception:
            pass
        finally:
            self.release()

    def close(self):
        with self._lock:
            if self.stream:
                try: 
                    self.stream.close()
                except Exception: 
                    pass
                self.stream = None
        super().close()


class CustomQueueListener(logging.handlers.QueueListener):
    """Listener que permite inyectar y remover handlers asíncronos en caliente."""
    def __init__(self, queue, *handlers, respect_handler_level=False):
        super().__init__(queue, *handlers, respect_handler_level=respect_handler_level)
        self._handlers_list = list(handlers)

    @property
    def handlers(self):
        return self._handlers_list

    @handlers.setter
    def handlers(self, value):
        self._handlers_list = list(value)

    def add_handler(self, handler):
        if handler not in self._handlers_list:
            self._handlers_list.append(handler)

    def _handle_one_record(self, record):
        if self._handlers_list:
            min_level = min(h.level for h in self._handlers_list)
            if record.levelno >= min_level:
                for handler in self._handlers_list:
                    if record.levelno >= handler.level:
                        handler.handle(record)


class LogManager:
    """Administrador único de logs asíncronos (Patrón Singleton)."""
    _instance: Optional['LogManager'] = None
    _loggers: Dict[str, logging.Logger] = {}
    _initialized: bool = False
    _listener: Optional[CustomQueueListener] = None
    _queue: Optional[queue.Queue] = None
    _file_handlers: List[DailyFolderHandler] = []

    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
        return cls._instance

    def __init__(self):
        if self._initialized:
            return
        self.base_dir = LOGS_MASTER_DIR
        self.critical_dir = LOGS_CRITICAL_DIR
        self._setup_root_configuration()
        self._initialized = True

    def _setup_root_configuration(self):
        self._queue = queue.Queue(maxsize=100000)
        root = logging.getLogger()
        root.setLevel(logging.DEBUG)

        for handler in root.handlers[:]:
            root.removeHandler(handler)

        formatter = logging.Formatter(
            '%(asctime)s | %(name)-25s | %(levelname)-8s | %(message)s',
            datefmt='%Y-%m-%d %H:%M:%S'
        )

        system_handler = DailyFolderHandler(self.base_dir, "00_ORCHESTRATOR", "system_main.log")
        system_handler.setLevel(logging.INFO)
        system_handler.setFormatter(formatter)
        system_handler.addFilter(logging.Filter("00_ORCHESTRATOR"))
        self._file_handlers.append(system_handler)

        critical_handler = DailyFolderHandler(self.base_dir, "CRITICAL", "system_fatal.log")
        critical_handler.setLevel(logging.ERROR)
        critical_handler.setFormatter(formatter)
        self._file_handlers.append(critical_handler)

        queue_handler = logging.handlers.QueueHandler(self._queue)
        queue_handler.setLevel(logging.DEBUG)
        root.addHandler(queue_handler)

        self._listener = CustomQueueListener(
            self._queue,
            system_handler,
            critical_handler,
            respect_handler_level=True
        )
        self._listener.start()

    def get_logger(self, name: str, category: str = "00_ORCHESTRATOR") -> logging.Logger:
        legacy_map = {
            "system": "00_ORCHESTRATOR",
            "capturers": "01_CAPTURE",
            "performance": "01_CAPTURE",
            "alerts": "00_ORCHESTRATOR",
            "errors": "CRITICAL",
            "01_CAPTURE_ENGINE": "01_CAPTURE",
            "02_AUDIT_ENGINE": "02_AUDIT",
            "03_QUANT_ENGINE": "03_QUANT",
            "04_REPORT_ENGINE": "04_REPORT"
        }
        category = legacy_map.get(category, category)
        
        valid_categories = ["01_CAPTURE", "02_AUDIT", "03_QUANT", "04_REPORT", "00_ORCHESTRATOR", "CRITICAL"]
        if category not in valid_categories:
            category = "00_ORCHESTRATOR"

        logger_key = f"{category}.{name}"
        if logger_key in self._loggers:
            return self._loggers[logger_key]

        logger = logging.getLogger(logger_key)
        logger.setLevel(logging.DEBUG)
        logger.propagate = False
        
        for handler in logger.handlers[:]:
            logger.removeHandler(handler)

        formatter = logging.Formatter(
            '%(asctime)s | %(name)-25s | %(levelname)-8s | %(message)s',
            datefmt='%Y-%m-%d %H:%M:%S'
        )

        file_handler = DailyFolderHandler(self.base_dir, category, f"{name}.log")
        file_handler.setLevel(logging.DEBUG)
        file_handler.setFormatter(formatter)
        file_handler.addFilter(logging.Filter(logger_key))
        self._file_handlers.append(file_handler)

        if self._listener:
            self._listener.add_handler(file_handler)

        queue_handler = logging.handlers.QueueHandler(self._queue)
        queue_handler.setLevel(logging.DEBUG)
        logger.addHandler(queue_handler)

        # Si no está en VPS remoto, replicar salida estándar en consola para debug local
        if ENTORNO_ACTUAL != "VPS":
            console_handler = logging.StreamHandler(sys.stdout)
            console_handler.setLevel(logging.INFO)
            console_handler.setFormatter(formatter)
            logger.addHandler(console_handler)

        self._loggers[logger_key] = logger
        return logger

    def shutdown(self):
        if self._listener:
            self._listener.stop()
        for handler in self._file_handlers:
            try:
                handler.close()
            except Exception:
                pass

_log_manager_instance: Optional[LogManager] = None

def get_log_manager() -> LogManager:
    global _log_manager_instance
    if _log_manager_instance is None:
        _log_manager_instance = LogManager()
    return _log_manager_instance

def get_logger(name: str, category: str = "00_ORCHESTRATOR") -> logging.Logger:
    return get_log_manager().get_logger(name, category)

def get_capturer_logger(symbol: str) -> logging.Logger:
    return get_log_manager().get_logger(symbol.upper(), "01_CAPTURE")

def get_system_logger(component: str) -> logging.Logger:
    return get_log_manager().get_logger(component, "00_ORCHESTRATOR")

def get_audit_logger(component: str) -> logging.Logger:
    return get_log_manager().get_logger(component, "02_AUDIT")

def get_ops_logger(component: str) -> logging.Logger:
    return get_log_manager().get_logger(component, "03_QUANT")
