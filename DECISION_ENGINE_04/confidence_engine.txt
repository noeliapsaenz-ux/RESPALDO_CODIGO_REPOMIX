"""
==========================================================================
CONFIDENCE_ENGINE - Fusión de confianza global (Fail-Closed)
==========================================================================
Ubicación: DECISION_ENGINE_04/confidence_engine.py
Dependencias: contract_manager, numpy
Responsabilidad: Fusionar las 4 señales críticas (Edge, FCI, Anomaly,
                 Readiness) utilizando los pesos parametrizados de forma
                 defensiva y robusta ante pérdidas de datos.
==========================================================================
"""
import sys
from pathlib import Path
root_path = Path(__file__).resolve().parents[1]
if str(root_path) not in sys.path: sys.path.insert(0, str(root_path))

import argparse
import numpy as np
from datetime import datetime, timezone
from typing import Dict, Any, Optional

from SYSTEM_CANONICAL.system_sessions import BROKER_TZ
from SYSTEM_CANONICAL.system_symbols import get_symbols
from SYSTEM_CANONICAL.system_canonical import SymbolCaptureStatus
from SYSTEM_CANONICAL.system_states import CONFIDENCE_WEIGHTS, DECISION_THRESHOLDS
from SYSTEM_SERVICES.system_logger import get_system_logger, stop_logger_listener
from SYSTEM_SERVICES.system_paths import (
    AUDIT_REPORTS, DECISION_FEATURE_VECTORS, DECISION_EDGE_SCORES, 
    DECISION_CONFIDENCE_SCORES, QUANT_READINESS, QUANT_ANOMALIES
)
from SYSTEM_CONTRACTS.contract_manager import load_contract_safe, save_contract_atomic

logger = get_system_logger("confidence_engine", "04_DECISION")

def load_feature_vector(symbol: str, fecha: str) -> Optional[Dict[str, Any]]:
    fv_file = DECISION_FEATURE_VECTORS / f"fv_{symbol.upper()}_{fecha}.json"
    return load_contract_safe(fv_file) if fv_file.exists() else None

def load_edge_scores(fecha: str) -> Optional[Dict[str, Any]]:
    edge_file = DECISION_EDGE_SCORES / f"edge_scores_{fecha}.json"
    return load_contract_safe(edge_file) if edge_file.exists() else None

def load_readiness_report(fecha: str) -> Dict[str, Any]:
    readiness_file = QUANT_READINESS / f"readiness_report_{fecha}.json"
    return load_contract_safe(readiness_file) if readiness_file.exists() else {}

def load_anomaly_report(fecha: str) -> Dict[str, Any]:
    anomaly_file = QUANT_ANOMALIES / f"anomaly_report_{fecha}.json"
    return load_contract_safe(anomaly_file) if anomaly_file.exists() else {}

def get_fci_for_symbol(fecha: str, symbol: str) -> float:
    audit_file = AUDIT_REPORTS / f"audit_report_{fecha}.json"
    if not audit_file.exists():
        audit_file = AUDIT_REPORTS / f"audit_certificate_{fecha}.json"
        if not audit_file.exists(): return 0.0
    try:
        data = load_contract_safe(audit_file)
        for item in data.get("resultados_auditoria", data.get("certificados_por_simbolo", [])):
            if item.get("symbol") == symbol.upper():
                metadata = item.get("metadata", {})
                return float(metadata.get("fci_captura", item.get("fci_score", 0.0)))
        return 0.0
    except Exception: return 0.0

def get_anomaly_score_for_symbol(anomaly_report: Dict, symbol: str) -> tuple:
    for res in anomaly_report.get("results", []):
        if res.get("symbol") == symbol.upper():
            return float(res.get("anomaly_score", 0.0)), res.get("classification_level", "NORMAL")
    return 0.0, "NORMAL"

def get_readiness_for_symbol(readiness_report: Dict, symbol: str, style: str) -> tuple:
    for item in readiness_report.get("resultados_readiness", []):
        if item.get("symbol") == symbol.upper():
            style_data = item.get("readiness_by_style", {}).get(style, {})
            return float(style_data.get("readiness_score", 0.0)), style_data.get("status", "VETOED")
    return 0.0, "VETOED"

def calculate_confidence_score(edge: float, fci: float, anomaly: float, readiness: float, status: str) -> Dict[str, Any]:
    if status == SymbolCaptureStatus.NO_CONFIABLE.value:
        return {"decision_confidence_score": 0.0, "veto_applied": True, "veto_reason": "NO_CONFIABLE", "inputs": {}}
    
    weights = CONFIDENCE_WEIGHTS
    anomaly_inverted = 100.0 - anomaly
    confidence = (edge * weights['edge_score'] + fci * weights['fci'] + anomaly_inverted * weights['anomaly_score'] + readiness * weights['readiness'])
    confidence = float(np.clip(confidence, 0.0, 100.0))
    return {
        "decision_confidence_score": round(confidence, 1),
        "veto_applied": False,
        "inputs": {"execution_feasibility_score": round(edge, 1), "fci": round(fci, 1), "anomaly_score": round(anomaly, 1), "readiness": round(readiness, 1)}
    }

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--date", type=str, default="latest")
    args = parser.parse_args()
    fecha = args.date if args.date != "latest" else datetime.now(BROKER_TZ).strftime("%Y%m%d")
    
    edge_data = load_edge_scores(fecha)
    readiness_report = load_readiness_report(fecha)
    anomaly_report = load_anomaly_report(fecha)
    
    confidence_data = {"fecha": fecha, "timestamp_generacion": datetime.now(timezone.utc).isoformat(), "confidence_scores": {}}
    
    for symbol in get_symbols():
        fv = load_feature_vector(symbol, fecha)
        if fv is None: continue
        capture_status = fv.get("symbol_capture_status", SymbolCaptureStatus.NO_CONFIABLE.value)
        fci = get_fci_for_symbol(fecha, symbol)
        anomaly_score, anomaly_level = get_anomaly_score_for_symbol(anomaly_report, symbol)
        
        symbol_confidences = {}
        for style in ['scalping', 'intraday', 'swing']:
            edge_info = edge_data.get("edge_scores", {}).get(symbol, {}).get(style, {})
            edge_score = edge_info.get("edge_score", 0.0)
            readiness, readiness_status = get_readiness_for_symbol(readiness_report, symbol, style)
            
            result = calculate_confidence_score(edge_score, fci, anomaly_score, readiness, capture_status)
            symbol_confidences[style] = {**result, "anomaly_classification": anomaly_level, "readiness_status": readiness_status}
            if result["decision_confidence_score"] >= DECISION_THRESHOLDS["min_confidence_score"]:
                logger.info(f"✅ {symbol} | {style} | Conf: {result['decision_confidence_score']}")
        confidence_data["confidence_scores"][symbol] = symbol_confidences
    
    save_contract_atomic(confidence_data, DECISION_CONFIDENCE_SCORES / f"confidence_scores_{fecha}.json")
    stop_logger_listener()

if __name__ == "__main__": 
    main()
