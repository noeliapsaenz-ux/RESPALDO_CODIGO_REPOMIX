"""
==========================================================================
DECISION_CONTRACT - Generador de contratos de decisión
==========================================================================
Ubicación: DECISION_ENGINE_04/decision_contract.py
Dependencias: contract_manager, decision_contracts
Responsabilidad: Generar el veredicto definitivo ALLOW/DENY por símbolo y estilo
                 para la sesión siguiente, aplicando vetos duros si se detecta
                 falta de fiabilidad del feed (F2) o fallo de validación matemática (F3.5).
==========================================================================
"""
import sys
from pathlib import Path

root_path = Path(__file__).resolve().parents[1]
if str(root_path) not in sys.path:
    sys.path.insert(0, str(root_path))

import argparse
from datetime import datetime, timezone
from typing import Dict, Any, List

from SYSTEM_CANONICAL.system_states import DecisionStatus, ContractType, DECISION_THRESHOLDS
from SYSTEM_CANONICAL.system_sessions import BROKER_TZ
from SYSTEM_CANONICAL.system_symbols import get_symbols
from SYSTEM_SERVICES.system_logger import get_system_logger
from SYSTEM_SERVICES.system_paths import (
    DECISION_CONFIDENCE_SCORES,
    DECISION_CONTRACTS,
    DECISION_HYPOTHESES
)
from SYSTEM_CONTRACTS.contract_manager import load_contract_safe, save_contract_atomic
from decision_contracts import create_canonical_execution_contract, create_decision_symbol_entry

logger = get_system_logger("decision_contract", "04_DECISION")

def load_confidence_scores(fecha: str) -> Dict[str, Any]:
    path = DECISION_CONFIDENCE_SCORES / f"confidence_scores_{fecha}.json"
    return load_contract_safe(path) if path.exists() else {}

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--date", default="latest")
    args = parser.parse_args()
    fecha = args.date if args.date != "latest" else datetime.now(BROKER_TZ).strftime("%Y%m%d")

    logger.info(f"📋 Generando contrato de ejecución para {fecha}...")

    confidence_data = load_confidence_scores(fecha)
    if not confidence_data:
        logger.critical("❌ No se encontraron confidence_scores. Contrato con todos los símbolos en PASS.")
        decisions = []
        for symbol in get_symbols():
            decisions.append(
                create_decision_symbol_entry(
                    symbol=symbol,
                    decision_status=DecisionStatus.PASS.value,
                    execution_feasibility_score=0.0,
                    hypothesis_significance=0.0,
                    decision_confidence_score=0.0,
                    veto_reasons=["Falta confidence_scores"]
                )
            )
        contract = create_canonical_execution_contract(fecha, decisions, global_veto_active=True)
        save_contract_atomic(contract, DECISION_CONTRACTS / f"decision_contracts_{fecha}.json")
        logger.info("✅ Contrato de emergencia guardado (todos PASS).")
        return

    decisions = []
    for symbol, styles in confidence_data.get("confidence_scores", {}).items():
        for style, data in styles.items():
            score = data.get("decision_confidence_score", 0.0)
            # Determinar estado usando los enums correctos
            if data.get("veto_applied", False):
                status = DecisionStatus.PASS.value
                reason = data.get("veto_reason", "Veto activo")
            elif score >= DECISION_THRESHOLDS["min_confidence_score"]:
                status = DecisionStatus.EXECUTE.value
                reason = None
            else:
                status = DecisionStatus.PASS.value
                reason = f"Confianza insuficiente: {score:.1f}"

            decisions.append(
                create_decision_symbol_entry(
                    symbol=symbol,
                    decision_status=status,
                    execution_feasibility_score=data.get("inputs", {}).get("execution_feasibility_score", 0.0),
                    hypothesis_significance=0.0,  # Podrías extraer de hypotheses si quieres
                    decision_confidence_score=score,
                    veto_reasons=[reason] if reason else []
                )
            )

    global_veto = any(d["decision_status"] == DecisionStatus.PASS.value for d in decisions)
    contract = create_canonical_execution_contract(fecha, decisions, global_veto_active=global_veto)
    output_file = DECISION_CONTRACTS / f"decision_contracts_{fecha}.json"
    save_contract_atomic(contract, output_file)
    logger.info(f"✅ Contrato de ejecución guardado: {output_file.name}")

if __name__ == "__main__":
    main()
