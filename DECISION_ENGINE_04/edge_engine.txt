logger = get_system_logger("edge_engine", "04_DECISION")

def normalize_metric(metric_name: str, value: float) -> float:
    cap = EDGE_NORMALIZATION_CAPS.get(metric_name, 100.0)
    # INVERTED_METRICS logic implicitly handled if metric in specific list, 
    # but for simplicity keep original logic with new caps
    if metric_name in {'regime_transitions_count', 'liquidity_changes_count'}:
        inverted = max(0.0, cap - value)
        return float(np.clip((inverted / cap) * 100.0, 0.0, 100.0))
    return float(np.clip((min(value, cap) / cap) * 100.0, 0.0, 100.0))

def calculate_edge_score_for_style(fv: Dict[str, Any], style: str) -> Dict[str, Any]:
    weights = READINESS_STYLE_WEIGHTS.get(style)
    if weights is None: return {"edge_score": 0.0}
    dimension_scores = {}
    for dimension, dim_weight in weights.items():
        if dim_weight == 0.0: continue
        if dimension == 'persistence':
            dimension_scores[dimension] = (normalize_metric('regime_duration_bars', fv.get('regime_duration_bars', 1.0)) * 0.60) + \
                                          (normalize_metric('regime_age_minutes', fv.get('regime_age_minutes', 0.0)) * 0.40)
        elif dimension == 'execution':
            dimension_scores[dimension] = (normalize_metric('regime_transitions_count', fv.get('regime_transitions_count', 0.0)) * 0.50) + \
                                          (normalize_metric('liquidity_changes_count', fv.get('liquidity_changes_count', 0.0)) * 0.50)
        elif dimension == 'anomaly': dimension_scores[dimension] = normalize_metric('minutes_since_last_anomalous_minute', fv.get('minutes_since_last_anomalous_minute', 9999.0))
        elif dimension == 'spread': dimension_scores[dimension] = normalize_metric('minutes_since_extreme_spread', fv.get('minutes_since_extreme_spread', 9999.0))
        elif dimension == 'volatility':
            vol = fv.get('realized_volatility', 0.0)
            if vol < VOLATILITY_THRESHOLDS['QUIET']: dimension_scores[dimension] = EDGE_VOLATILITY_SCORES["quiet"]
            elif vol < VOLATILITY_THRESHOLDS['NORMAL']: dimension_scores[dimension] = EDGE_VOLATILITY_SCORES["normal"]
            elif vol < VOLATILITY_THRESHOLDS['ACTIVE']: dimension_scores[dimension] = EDGE_VOLATILITY_SCORES["active"]
            else: dimension_scores[dimension] = EDGE_VOLATILITY_SCORES["explosive"]
    
    score = sum(dimension_scores.get(dim, 0.0) * weights[dim] for dim in weights if dim in dimension_scores)
    return {"edge_score": round(float(np.clip(score, 0.0, 100.0)), 1), "breakdown": {k: round(v, 1) for k, v in dimension_scores.items()}}

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--date", default="latest")
    args = parser.parse_args()
    fecha = args.date if args.date != "latest" else datetime.now(BROKER_TZ).strftime("%Y%m%d")
    
    edge_data = {"fecha": fecha, "edge_scores": {}}
    for symbol in get_symbols():
        fv = load_contract_safe(DECISION_FEATURE_VECTORS / f"fv_{symbol.upper()}_{fecha}.json")
        if fv is None: continue
        symbol_edges = {}
        for style in ['scalping', 'intraday', 'swing']:
            res = calculate_edge_score_for_style(fv, style)
            symbol_edges[style] = res
            if res["edge_score"] >= READINESS_THRESHOLDS.get(f"min_score_{style}", 60.0):
                logger.info(f"✅ {symbol} | {style} | Edge: {res['edge_score']}")
        edge_data["edge_scores"][symbol] = symbol_edges
    save_contract_atomic(edge_data, DECISION_EDGE_SCORES / f"edge_scores_{fecha}.json")

if __name__ == "__main__": 
    main()
