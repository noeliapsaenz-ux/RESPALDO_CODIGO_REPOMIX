import sys
from pathlib import Path

root_path = Path(__file__).resolve().parents[1]
if str(root_path) not in sys.path:
    sys.path.insert(0, str(root_path))

from datetime import datetime, timezone
from typing import Dict, Any, Optional, List

# Importaciones canónicas unificadas (sin decision_config)
from SYSTEM_CANONICAL.system_states import DecisionStatus, ContractType
from SYSTEM_CANONICAL.system_canonical import MetadataEnvelope, validate_canonical_enum

# Definiciones de versión del esquema según contract_manager.py
SCHEMA_VERSION_DECISION = "3.0"
PRODUCER_DECISION = "DECISION_ENGINE_04"

def create_decision_symbol_entry(
    symbol: str,
    decision_status: str,
    execution_feasibility_score: float,
    hypothesis_significance: float,
    decision_confidence_score: float,
    veto_reasons: Optional[List[str]] = None
) -> Dict[str, Any]:
    if not validate_canonical_enum(decision_status, DecisionStatus):
        raise ValueError(f"decision_status inválido: {decision_status}")

    return {
        "symbol": symbol.upper(),
        "decision_status": decision_status,
        "execution_feasibility_score": round(float(execution_feasibility_score), 4),
        "hypothesis_significance": round(float(hypothesis_significance), 4),
        "decision_confidence_score": round(float(decision_confidence_score), 4),
        "veto_reasons": veto_reasons or []
    }

def create_canonical_execution_contract(
    fecha: str,
    decisions: List[Dict[str, Any]],
    global_veto_active: bool = False
) -> Dict[str, Any]:
    envelope = MetadataEnvelope(
        schema_version=SCHEMA_VERSION_DECISION,
        contract_type=ContractType.EXECUTION,
        producer_identity=PRODUCER_DECISION,
        timestamp_emission=datetime.now(timezone.utc).isoformat()
    )

    return {
        **envelope.to_dict(),
        "fecha": fecha,
        "global_veto_active": global_veto_active,
        "decisions": decisions
    }

def validate_execution_contract(contract: Dict[str, Any]) -> bool:
    required = {"schema_version", "contract_type", "producer_identity",
                "timestamp_emission", "fecha", "decisions"}
    if not required.issubset(contract.keys()):
        return False
    if contract.get("contract_type") != ContractType.EXECUTION.value:
        return False
    return True
