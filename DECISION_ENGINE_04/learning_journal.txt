def load_previous_day_contracts(fecha: str) -> List[Dict[str, Any]]:
    prev_date = (datetime.strptime(fecha, "%Y%m%d") - timedelta(days=1)).strftime("%Y%m%d")
    contract_file = DECISION_WORKSPACE_DIR / "contracts" / f"decision_contracts_{prev_date}.json"
    if not contract_file.exists(): 
        return []
    try:
        with open(contract_file, "r", encoding="utf-8") as f: 
            data = json.load(f)
        return [c for c in data.get("contracts", []) if c.get("decision") == "ALLOW"]
    except Exception: 
        return []

def load_market_state(symbol: str, fecha: str) -> Optional[pd.DataFrame]:
    # Corregido: Se redirige la busqueda de la carpeta legada "market_state" a la carpeta real "bars"
    state_file = PROCESSED_DATA_DIR / "bars" / symbol / f"{symbol}_{fecha}_market_state.parquet"
    if not state_file.exists(): 
        return None
    try: 
        return pd.read_parquet(state_file)
    except Exception: 
        return None
