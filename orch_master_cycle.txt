background_processes: List[subprocess.Popen] = []


def launch_python_script(folder: Optional[str], script_name: str, args: List[str] = None,
                         background: bool = False) -> Optional[subprocess.Popen]:
    """
    Lanza un script Python en foreground o background.

    Args:
        folder: Carpeta donde se encuentra el script (None para raíz)
        script_name: Nombre del script
        args: Argumentos para el script
        background: True para ejecutar en background

    Returns:
        Optional[subprocess.Popen]: Proceso si es background, None si es foreground
    """
    script_path = (raiz_proyecto / folder / script_name) if folder else (raiz_proyecto / script_name)
    if not script_path.exists():
        logger.error(f"❌ Script no localizado: {script_path}")
        return None

    cmd = [sys.executable, "-u", str(script_path)] + (args or [])
    cwd_dir = raiz_proyecto / folder if folder else raiz_proyecto
    status = "background" if background else "foreground"
    logger.info(f"▶️ Lanzando {script_name} en [{status}]")

    try:
        if background:
            creation_flags = subprocess.CREATE_NEW_CONSOLE if sys.platform == "win32" else 0
            proc = subprocess.Popen(cmd, creationflags=creation_flags, cwd=str(cwd_dir), shell=False)
            logger.info(f"   ✅ Ventana de terminal abierta (PID: {proc.pid})")
            return proc
        else:
            logger.info(f"   ⏳ Ejecutando (bloqueando hilo principal)...")
            result = subprocess.run(cmd, cwd=str(cwd_dir), capture_output=False, text=True, encoding='utf-8')
            if result.returncode == 0:
                logger.info(f"   ✅ Finalizado correctamente")
            else:
                logger.error(f"   ❌ Finalizado con código de error: {result.returncode}")
            return None
    except Exception as e:
        logger.error(f"❌ Error lanzando {script_name}: {e}")
        return None


def stop_background_processes():
    """Detiene todos los procesos en background."""
    global background_processes
    if not background_processes:
        return

    logger.info("🛑 Deteniendo procesos de soporte en background...")
    for proc in background_processes:
        if proc and proc.poll() is None:
            try:
                proc.terminate()
                proc.wait(timeout=5)
                logger.info(f"   ✅ Proceso {proc.pid} finalizado.")
            except subprocess.TimeoutExpired:
                proc.kill()
                logger.warning(f"   ⚠️ Proceso {proc.pid} forzado (kill).")
            except Exception as e:
                logger.error(f"   ❌ Error al detener proceso {proc.pid}: {e}")
    background_processes.clear()


def launch_capture_phase() -> bool:
    """
    Lanza la fase de captura HFT diaria.

    Returns:
        bool: True si la captura se inició correctamente
    """
    global background_processes

    logger.info("=" * 80)
    logger.info("🚀 INICIANDO FASE DE CAPTURA HFT DIARIA")
    logger.info("=" * 80)

    # Limpiar archivos de estado previos
    for sym in SYMBOLS_TO_CAPTURE:
        (STATE_DIR / f"{sym}.shutdown_ok").unlink(missing_ok=True)
    (STATE_DIR / "shutdown.signal").unlink(missing_ok=True)

    # 1. Lanzar Compuerta Forense
    logger.info("\n📦 [1/3] Lanzando Compuerta Forense (archive_daemon.py - modo demonio)")
    proc_daemon = launch_python_script("CAPTURE_ENGINE_01", "archive_daemon.py", background=True)
    if proc_daemon:
        background_processes.append(proc_daemon)

    time.sleep(2)

    # 2. Lanzar Panel de Telemetría
    logger.info("\n📊 [2/3] Lanzando Panel de Telemetría (telemetry_dashboard.py)")
    proc_dash = launch_python_script("CAPTURE_ENGINE_01", "telemetry_dashboard.py", background=True)
    if proc_dash:
        background_processes.append(proc_dash)

    time.sleep(2)

    # 3. Iniciar Supervisor de Captura
    logger.info("\n🔄 [3/3] Iniciando Supervisor de Captura (orch_live_capture.py)")
    launch_python_script(None, "orch_live_capture.py", background=False)

    logger.info("=" * 80)
    logger.info("✅ SUPERVISOR DE CAPTURA DIARIO CONCLUIDO")
    logger.info("=" * 80)
    return True


def get_closed_session_date() -> str:
    """
    Obtiene la fecha de la sesión cerrada más reciente.

    Returns:
        str: Fecha en formato YYYYMMDD
    """
    now_broker = datetime.now(BROKER_TZ)
    if now_broker.hour < 2:
        return (now_broker - timedelta(days=1)).strftime("%Y%m%d")
    return now_broker.strftime("%Y%m%d")

    """
    Ejecuta la secuencia nocturna completa (F2, F3, F4).

    Args:
        fecha: Fecha en formato YYYYMMDD
    """
    logger.info("=" * 80)
    logger.info(f"🌙 INICIANDO SECUENCIA NOCTURNA COMPLETA: {fecha}")
    logger.info("=" * 80)

    # 1. Forzar vaciado final de buffers HFT
    logger.info("\n📦 [CUSTODIA] Vaciado de Buffers HFT (single-pass)")
    launch_python_script("CAPTURE_ENGINE_01", "archive_daemon.py", ["--single-pass"])

    # 2. Ejecutar Orquestador de Auditoría (Fase 2)
    logger.info("\n⚖️ [AUDITORÍA] Lanzando Orquestador de Fase 2 (orch_audit.py)")
    launch_python_script(None, "orch_audit.py", ["--date", fecha, "--send-telegram"])

    # 3. Ejecutar Orquestador de Inteligencia Operacional (Fase 3)
    logger.info("\n🧠 [INTELIGENCIA] Lanzando Orquestador de Fase 3 (orch_intelligence.py)")
    launch_python_script(None, "orch_intelligence.py", ["--date", fecha, "--send-telegram"])

    # 4. Ejecutar Orquestador de Decisiones (Fase 4)
    logger.info("\n🎯 [DECISIÓN] Lanzando Orquestador de Fase 4 (orch_decision.py)")
    launch_python_script(None, "orch_decision.py", ["--date", fecha])


def enviar_notificacion_telegram_consolidada(fecha: str, start_time: str, stop_time: str):
    """
    Envía notificación consolidada por Telegram con resumen del ciclo.

    Args:
        fecha: Fecha en formato YYYYMMDD
        start_time: Hora de inicio de la captura
        stop_time: Hora de fin de la captura
    """
    f2_docx = AUDIT_REPORTES_DIR / f"diagnostico_operacional_{fecha}.docx"
    f3_docx = OI_OUTPUT_DIR / f"diagnostico_operacional_F3_{fecha}.docx"

    contract_file = AUDIT_STATE_DIR / "portfolio_execution_contract.json"

    assets_sc, assets_id, assets_sw = "FILTRADO", "FILTRADO", "FILTRADO"
    if contract_file.exists():
        try:
            with open(contract_file, "r") as f:
                data = json.load(f)
            styles = data.get("readiness_by_style", {})
            sel_sc = styles.get("scalping", {})
            if sel_sc.get("status") == "OPERABLE":
                assets_sc = ", ".join(sel_sc.get("trio", []))
            sel_id = styles.get("intraday", {})
            if sel_id.get("status") == "OPERABLE":
                assets_id = ", ".join(sel_id.get("trio", []))
            sel_sw = styles.get("swing", {})
            if sel_sw.get("status") == "OPERABLE":
                assets_sw = ", ".join(sel_sw.get("trio", []))
        except Exception:
            pass

    msg_lines = [
        f"🌙 *CICLO DIARIO COMPLETADO — {fecha}*",
        f"",
        f"📊 *Resumen de Ejecución:*",
        f"   • Captura: {start_time} → {stop_time}",
        f"   • Reporte F2: {'✅' if f2_docx.exists() else '❌'}",
        f"   • Reporte F3: {'✅' if f3_docx.exists() else '❌'}",
        f"",
        f"🎯 *Activos Operables:*",
        f"   • Scalping: {assets_sc}",
        f"   • Intraday: {assets_id}",
        f"   • Swing: {assets_sw}",
    ]
    send_telegram_alert("\n".join(msg_lines))


def get_seconds_until_next_session() -> float:
    """
    Calcula los segundos hasta la próxima sesión de mercado.

    Returns:
        float: Segundos hasta la próxima sesión
    """
    now = datetime.now(BROKER_TZ)
    weekday = now.weekday()

    # Si es viernes (4), esperar hasta el lunes (2 días)
    # Si es sábado (5), esperar hasta el lunes (1 día)
    # Si es domingo (6), esperar hasta el lunes (1 día)
    days_to_add = 3 if weekday == 4 else (2 if weekday == 5 else 1)

    next_session = now.replace(hour=0, minute=0, second=0, microsecond=0) + timedelta(days=days_to_add)
    return (next_session - now).total_seconds()


def signal_handler(sig, frame):
    """Maneja la señal de interrupción (Ctrl+C)."""
    logger.info("\n🛑 Interrupción capturada. Deteniendo subprocesos...")
    stop_background_processes()
    sys.exit(0)


def main():
    parser = argparse.ArgumentParser(description="Orch Master Cycle - Orquestador del Ciclo de Vida Diario")
    parser.add_argument("--test-only", action="store_true",
                       help="Ejecutar solo secuencia nocturna F2+F3 y salir")
    parser.add_argument("--skip-capture", action="store_true",
                       help="Saltar fase de captura y proceder a post-market")
    args = parser.parse_args()

    signal.signal(signal.SIGINT, signal_handler)

    # Modo prueba: ejecutar solo la secuencia nocturna
    if args.test_only:
        fecha_sesion = get_closed_session_date()
        logger.info(f"🧪 [MODO PRUEBA] Ejecución acelerada para: {fecha_sesion}")
        run_post_market_processing(fecha_sesion)
        enviar_notificacion_telegram_consolidada(fecha_sesion, "Simulado", "Simulado")
        logger.info("🧪 [MODO PRUEBA] Secuencia finalizada.")
        return

    # Modo normal: ciclo completo
    logger.info("=" * 80)
    logger.info("🚀 MAESTRO DE CICLO DE VIDA OPERACIONAL MT5")
    logger.info("=" * 80)

    while True:
        try:
            ahora_broker = datetime.now(BROKER_TZ)
            logger.info(f"🔄 Sincronizando reloj operacional. Hora Broker: {ahora_broker.strftime('%Y-%m-%d %H:%M:%S')}")

            # Verificar si el mercado está cerrado (fin de semana)
            if ahora_broker.weekday() in [5, 6] and not is_market_open():
                sleep_sec = get_seconds_until_next_session()
                logger.info(f"⏸️ Mercado en receso. Reposo por {sleep_sec/3600:.1f} horas...")
                time.sleep(sleep_sec)
                continue

            # Iniciar captura
            start_capture_time = datetime.now(BROKER_TZ).strftime('%H:%M:%S')

            if not args.skip_capture:
                send_telegram_alert(
                    f"🚀 *INICIO DE SESIÓN DE MERCADO — {ahora_broker.strftime('%Y%m%d')}*\n"
                    f"Adquisición de ticks iniciada."
                )
                launch_capture_phase()
                stop_background_processes()
            else:
                logger.info("⏭️ [SKIP] Saltando fase de captura activa.")

            stop_capture_time = datetime.now(BROKER_TZ).strftime('%H:%M:%S')

            # Ejecutar post-market
            fecha_sesion = get_closed_session_date()
            run_post_market_processing(fecha_sesion)

            # Enviar notificación consolidada
            enviar_notificacion_telegram_consolidada(fecha_sesion, start_capture_time, stop_capture_time)

            # Limpiar archivos de estado
            for sym in SYMBOLS_TO_CAPTURE:
                (STATE_DIR / f"{sym}.shutdown_ok").unlink(missing_ok=True)

            # Esperar hasta la próxima sesión
            sleep_sec = get_seconds_until_next_session()
            logger.info(f"✅ Ciclo diario completado. Durmiendo por {sleep_sec/3600:.1f} horas...")
            time.sleep(sleep_sec)

        except KeyboardInterrupt:
            logger.info("\n🛑 Orquestador maestro detenido por el operador.")
            stop_background_processes()
            break
        except Exception as e:
            logger.error(f"❌ Error fatal en bucle orquestador maestro: {e}", exc_info=True)
            stop_background_processes()
            time.sleep(10)


if __name__ == "__main__":
    main()
