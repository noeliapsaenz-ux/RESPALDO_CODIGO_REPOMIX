def get_adaptive_timeout(fecha: str) -> int:
    """Calcula dinámicamente el tiempo límite de ejecución según el volumen en disco."""
    base_timeout = 600
    try:
        from paths import SSD_EVIDENCE_DIR
        total_bytes = 0
        if SSD_EVIDENCE_DIR.exists():
            for p in SSD_EVIDENCE_DIR.rglob(f"*{fecha}*"):
                if p.is_file():
                    total_bytes += p.stat().st_size
        size_mb = total_bytes / (1024 * 1024)
        adaptive_bonus = int(size_mb * 30)
        total_timeout = base_timeout + adaptive_bonus
        logger.info(f"⏱️ Timeout adaptativo calibrado en {total_timeout}s.")
        return total_timeout
    except Exception:
        return base_timeout


def ejecutar_script_pipeline(folder: str, script_name: str, fecha: str, timeout: int = 600) -> bool:
    script_path = raiz_proyecto / folder / script_name
    if not script_path.exists():
        logger.error(f"❌ Script no localizado: {script_path}")
        return False

    env_utf8 = os.environ.copy()
    env_utf8["PYTHONIOENCODING"] = "utf-8"
    env_utf8["PYTHONPATH"] = str(raiz_proyecto) + os.pathsep + env_utf8.get("PYTHONPATH", "")

    cmd = [sys.executable, "-u", str(script_path), "--date", fecha]
    logger.info(f"🔄 Corriendo: {script_name} ({fecha})...")
    
    try:
        result = subprocess.run(
            cmd, capture_output=True, text=True, encoding='utf-8', errors='replace', 
            timeout=timeout, cwd=str(raiz_proyecto / folder), env=env_utf8
        )
        if result.returncode == 0:
            logger.info(f"✅ {script_name} finalizado con éxito.")
            return True
        else:
            stderr_summary = result.stderr[-400:] if result.stderr else "Sin detalles."
            logger.error(f"❌ {script_name} devolvió código {result.returncode}.\nDetalle:\n{stderr_summary}")
            return False
    except subprocess.TimeoutExpired:
        logger.critical(f"🛑 TIMEOUT: {script_name} excedió el tiempo máximo de {timeout}s.")
        return False
    except Exception as e:
        logger.error(f"❌ Error inesperado ejecutando {script_name}: {e}")
        return False


def send_intelligence_summary_telegram(fecha: str, results: dict, elapsed: float):
    passed = sum(1 for v in results.values() if "PASS" in str(v))
    total = len(OPERATIONAL_PIPELINE)
    status_icon = "✅" if passed == total else "⚠️"
    
    msg_lines = [
        f"{status_icon} *FASE 3 — INTELIGENCIA COMPLETADA*",
        f"📅 Fecha: `{fecha}`",
        f"📊 *Resumen de ejecución:*",
        f"   • Módulos completados: {passed}/{total}",
        f"   • Tiempo total: {elapsed:.1f}s",
    ]
    send_telegram_alert("\n".join(msg_lines))


def main():
    parser = argparse.ArgumentParser(description="Operational Pipeline Orchestrator")
    parser.add_argument("--date", type=str, default="latest", help="Fecha AAAAMMDD")
    parser.add_argument("--send-telegram", action='store_true', help="Enviar resumen a Telegram al finalizar")
    args = parser.parse_args()
    
    fecha = args.date if args.date != "latest" else get_fecha_broker_str()
    
    logger.info("=" * 80)
    logger.info(f"🧠 INICIANDO PIPELINE DE INTELIGENCIA OPERACIONAL PARA: {fecha}")
    logger.info("=" * 80)
    
    start_time = time.time()
    results = {}
    abort_pipeline = False
    
    timeout = get_adaptive_timeout(fecha)
    
    for step in OPERATIONAL_PIPELINE:
        folder, name, desc, is_critical = step["folder"], step["file"], step["desc"], step["critical"]
        
        if abort_pipeline:
            logger.warning(f"⏩ OMITIDO: {name} ({desc}) debido a fallo crítico previo.")
            results[name] = "OMITIDO"
            continue

        step_start = time.time()
        success = ejecutar_script_pipeline(folder, name, fecha, timeout)
        step_elapsed = time.time() - step_start
        
        if success:
            results[name] = f"PASS ({step_elapsed:.1f}s)"
        else:
            results[name] = f"FAIL ({step_elapsed:.1f}s)"
            if is_critical:
                logger.critical(f"🛑 FALLO CRÍTICO en '{name}' ({desc}). El pipeline se detendrá.")
                abort_pipeline = True
                
    elapsed_total = time.time() - start_time
    passed_steps = sum(1 for v in results.values() if "PASS" in str(v))
    total_steps = len(OPERATIONAL_PIPELINE)
    
    logger.info("=" * 80)
    logger.info(f"📋 RESUMEN EJECUTIVO FASE 3 - FECHA: {fecha}")
    logger.info("=" * 80)
    for script, status in results.items():
        logger.info(f"  • {script:25} : {status}")
    logger.info("-" * 70)
    logger.info(f"⏱️ Tiempo total: {elapsed_total:.1f}s.")
    logger.info(f"📊 Pasos completados: {passed_steps}/{total_steps}")
    logger.info("=" * 80)

    if args.send_telegram:
        send_intelligence_summary_telegram(fecha, results, elapsed_total)


if __name__ == "__main__":
    main()
