    def _check_process_alive(self, symbol: str) -> Tuple[bool, str]:
        try:
            for proc in psutil.process_iter(['pid', 'name', 'cmdline']):
                try:
                    cmdline = ' '.join(proc.info['cmdline'] or [])
                    if ('mt5_tick_engine.py' in cmdline) and symbol.upper() in cmdline:
                        if proc.is_running(): return True, f"PID {proc.info['pid']}"
                except (psutil.NoSuchProcess, psutil.AccessDenied): continue
            return False, "No encontrado"
        except Exception: return False, "Error"

    def _check_heartbeat(self, symbol: str) -> Tuple[bool, float]:
        heartbeat_file = STATE_DIR / f"{symbol}.heartbeat"
        if not heartbeat_file.exists(): return False, 0.0
        try:
            with open(heartbeat_file, 'r', encoding='utf-8') as f:
                content = f.read().strip()
                lines = content.split('\n')
                if lines:
                    timestamp = float(lines[0].strip())
                    age = time.time() - timestamp
                    return True, age
        except Exception: pass
        return False, 0.0

    def _determine_real_state(self, symbol: str, data: Dict[str, Any]) -> Tuple[str, str, str]:
        process_alive, pid_info = self._check_process_alive(symbol)
        heartbeat_exists, heartbeat_age = self._check_heartbeat(symbol)
        system_timestamp = data.get('system_timestamp', 0)
        if system_timestamp > 0:
            data_age = time.time() - system_timestamp
            if data_age < 0: data_age = 0.0
        else:
            state_file = STATE_DIR / f"{symbol}_state.json"
            if state_file.exists(): data_age = time.time() - state_file.stat().st_mtime
            else: data_age = float('inf')
        
        data_fresh = data_age < 30.0
        data_critical = data_age > 120.0
        
        if process_alive: process_status = "RUNNING"
        else: process_status = "STOPPED"
        
        if process_status == "RUNNING" and data_fresh:
            real_status, real_color, detail = "LIVE", "GREEN", f"PID activo | Heartbeat: {heartbeat_age:.0f}s"
        elif process_status == "RUNNING" and not data_fresh and not data_critical:
            real_status, real_color, detail = "STALE", "YELLOW", f"Proceso vivo | Sin datos {data_age:.0f}s"
        elif process_status == "RUNNING" and data_critical:
            real_status, real_color, detail = "STALLED", "ORANGE", f"Proceso atascado | Sin datos {data_age:.0f}s"
        elif process_status == "STOPPED" and data_age > 60:
            real_status, real_color, detail = "DEAD", "RED", f"Proceso detenido | Último dato {data_age:.0f}s"
        else:
            real_status, real_color, detail = "OFFLINE", "GRAY", "Sistema inactivo"
        return real_status, real_color, detail

    def check_dead_processes(self) -> List[Tuple[str, int]]:
        dead_symbols = []
        for symbol, info in list(self.processes.items()):
            process = info.get('process')
            if process and process.poll() is not None: dead_symbols.append((symbol, process.poll()))
        return dead_symbols

    def check_zombie_processes(self) -> List[Tuple[str, float, Dict]]:
        zombies = []
        ahora = time.time()
        for symbol, info in list(self.processes.items()):
            process = info.get('process')
            if not process or process.poll() is not None: continue
            if not is_market_open(): continue
            is_zombie = False
            zombie_age = 0
            session = self._current_session.get(symbol, self._get_current_session())
            umbral_critico = get_session_umbral_critico(session)
            heartbeat_file = STATE_DIR / f"{symbol}.heartbeat"
            if heartbeat_file.exists():
                try:
                    with open(heartbeat_file, 'r', encoding='utf-8') as f:
                        content = f.read().strip()
                        last_beat = float(content.split('\n')[0])
                        heartbeat_age = ahora - last_beat
                        if heartbeat_age > umbral_critico:
                            is_zombie = True
                            zombie_age = heartbeat_age
                except Exception: pass
            if is_zombie: zombies.append((symbol, zombie_age, info))
        return zombies

    def start_capturer_smart(self, symbol: str) -> bool:
        print(f"{Fore.WHITE}▶️ Iniciando {symbol}... ", end=' ', flush=True)
        try:
            fecha_str = datetime.now(BROKER_TZ).strftime('%Y%m%d')
            log_dir = LOGS_MASTER_DIR / fecha_str / "01_CAPTURE"
            log_dir.mkdir(parents=True, exist_ok=True)
            log_path = log_dir / f"{symbol}_stderr.log"
            capturer_path = raiz_proyecto / "CAPTURE_ENGINE_01" / "mt5_tick_engine.py"
            cmd = [sys.executable, "-u", str(capturer_path), symbol]
            log_file = open(log_path, "a", encoding='utf-8', buffering=1)
            env_utf8 = os.environ.copy()
            env_utf8["PYTHONIOENCODING"] = "utf-8"
            env_utf8["PYTHONPATH"] = str(raiz_proyecto) + os.pathsep + env_utf8.get("PYTHONPATH", "")
            
            p = subprocess.Popen(cmd, stdout=subprocess.DEVNULL, stderr=log_file, cwd=str(raiz_proyecto), env=env_utf8)
            self.processes[symbol] = {
                'process': p, 'pid': p.pid, 'start_time': time.time(),
                'last_restart': datetime.now(), 'restart_count': self.restart_count.get(symbol, 0),
                'status': 'running', 'log_path': str(log_path), 'log_file': log_file
            }
            self._update_session_for_symbol(symbol)
            print(f"{Fore.GREEN}✅ PID: {p.pid}")
            self.logger.info(f"✅ {symbol} iniciado de forma asíncrona en CAPTURE_ENGINE_01 (PID: {p.pid})")
            return True
        except Exception as e:
            print(f"{Fore.RED}❌ Falló: {e}")
            self.problems_detected.append(f"{symbol}: {str(e)[:50]}")
            self.logger.error(f"❌ Error iniciando proceso capturador para {symbol}: {e}")
            return False

    def restart_dead_process(self, symbol: str, exit_code: int) -> bool:
        if symbol not in self.processes: return False
        tiempo_vivo = time.time() - self.processes[symbol]['start_time']
        old_log = self.processes[symbol].get('log_file')
        if old_log:
            try: old_log.close()
            except Exception: pass
        ahora = time.time()
        if symbol in self.pausado_hasta and ahora < self.pausado_hasta[symbol]: return False
        
        if exit_code == 0 and tiempo_vivo >= TIEMPO_VIDA_MINIMO_SEG:
            self.restart_count[symbol] = 0
            print(f"\n{Fore.CYAN}🔄 {symbol} finalizó de forma controlada — Re-iniciando...")
            return self.start_capturer_smart(symbol)
            
        reinicios = self.restart_count.get(symbol, 0) + 1
        self.restart_count[symbol] = reinicios
        if reinicios > MAX_REINICIOS:
            self.pausado_hasta[symbol] = ahora + BACKOFF_ESPERA_SEG
            self.restart_count[symbol] = 0
            msg = f"{symbol}: {MAX_REINICIOS} caídas detectadas — Captura pausada por {BACKOFF_ESPERA_SEG}s"
            print(f"\n{Fore.RED}⛔ {msg}")
            self.problems_detected.append(msg)
            send_telegram_alert(f"⛔ <b>{msg}</b>")
            self.logger.warning(f"⛔ {msg}")
            return False
            
        print(f"\n{Fore.YELLOW}💀 PROCESO CAÍDO: {symbol} (Exit: {exit_code}, Uptime: {tiempo_vivo:.1f}s)")
        print(f"{Fore.YELLOW} 🔄 Recuperando de forma asíncrona ({reinicios}/{MAX_REINICIOS})...")
        self.logger.warning(f"💀 {symbol} caído en sesión. Re-lanzando de forma controlada...")
        return self.start_capturer_smart(symbol)

    def _get_state_info(self, symbol: str) -> dict:
        try:
            state_file = STATE_DIR / f"{symbol}_system_state.json"
            if state_file.exists():
                with open(state_file, 'r', encoding='utf-8') as f:
                    data = json.load(f)
                    return data.get('metrics', {})
        except Exception as e:
            self.logger.debug(f"Error leyendo estado de {symbol}: {e}")
        return {}

            print(f"\n{Fore.CYAN}{'SÍMBOLO':<10} {'ESTADO SM':<10} {'VEL':<6} {'BUFFER':<14} {'OVERFLOW':<10} {'SLOPE':<8} {'REIN'}")
            print(f"{Fore.CYAN}{'-'*100}")

                info = self.processes.get(symbol, {})
                process = info.get('process')

                else:
                    status_icon = f"{Fore.RED}💀"
                    status_text = "DETENIDO"

                default_sm_state = SystemState.GREEN.value if status_text == "ACTIVO" else "OFFLINE"
                state = state_info.get('state', default_sm_state)

    def restart_all(self):
        print(f"\n{Fore.YELLOW}🔄 Forzando apagado y re-inicialización del pipeline...")
        self.logger.info("🔄 Re-inicialización manual del pipeline de captura.")
        self.shutdown_all_graceful()
        time.sleep(3)
        self.processes.clear()
        self.restart_count = {symbol: 0 for symbol in SYMBOLS_TO_CAPTURE}
        self.pausado_hasta.clear()
        for i, symbol in enumerate(SYMBOLS_TO_CAPTURE):
            self.start_capturer_smart(symbol)
            if i < len(SYMBOLS_TO_CAPTURE) - 1: time.sleep(DELAY_BETWEEN_STARTS)
        print(f"\n{Fore.GREEN}✅ Pipeline completo re-lanzado")
        self.logger.info("✅ Re-arranque del pipeline completado.")

    def start_all_capturers(self):
        print(f"\n{Fore.GREEN}🚀 LANZANDO CAPTURADORES INDEPENDIENTES...")
        for i, symbol in enumerate(SYMBOLS_TO_CAPTURE):
            self.start_capturer_smart(symbol)
            if i < len(SYMBOLS_TO_CAPTURE) - 1: time.sleep(DELAY_BETWEEN_STARTS)
        print(f"\n{Fore.GREEN}✅ {len(SYMBOLS_TO_CAPTURE)} capturadores inicializados.")
        print(f"{Fore.YELLOW}⏳ Esperando {STABILIZATION_TIMEOUT}s para que los capturadores escriban su primer estado...")
        time.sleep(STABILIZATION_TIMEOUT)
        for symbol in SYMBOLS_TO_CAPTURE: self._get_state_info(symbol)

    def _wait_for_market(self) -> bool:
        print(f"\n{Fore.YELLOW}⏸️ En espera de la apertura del mercado...")
        while self.running:
            try:
                if is_market_open():
                    self.market_open = True
                    print(f"\n{Fore.GREEN}🔔 ¡MERCADO ABIERTO!")
                    self.logger.info("🔔 Apertura de mercado detectada. Arrancando de forma coordinada.")
                    return True
                from SYSTEM_CANONICAL.system_sessions import get_current_broker_time
                ahora_broker = get_current_broker_time()
                print(f"\r{Fore.YELLOW}⏳ {ahora_broker.strftime('%H:%M:%S')} (Hora Broker en espera)", end='', flush=True)
                time.sleep(15)
            except KeyboardInterrupt:
                self.running = False
                return False
        return False

    def shutdown_all_graceful(self):
        print(f"\n{Fore.RED}🔴 ORQUESTANDO APAGADO CONTROLADO (SHUTDOWN SIGNAL)...")
        self.logger.info("🔴 Iniciando apagado seguro mediante archivo de señal en disco.")
        signal_file = STATE_DIR / "shutdown.signal"
        try:
            with open(signal_file, "w", encoding="utf-8") as f:
                f.write(f"shutdown_initiated_at={time.time()}\nby_pid={os.getpid()}\n")
            self.logger.info("✅ Señal de parada física 'shutdown.signal' creada en el disco.")
        except Exception as e:
            self.logger.error(f"Fallo crítico escribiendo señal en disco: {e}")
            self._fallback_force_kill()
            return
            
        start_wait = time.time()
        confirmed_symbols = set()
        while time.time() - start_wait < SHUTDOWN_GRACEFUL_WAIT:
            for symbol in SYMBOLS_TO_CAPTURE:
                if symbol not in confirmed_symbols:
                    ok_file = STATE_DIR / f"{symbol}.shutdown_ok"
                    if ok_file.exists():
                        confirmed_symbols.add(symbol)
                        print(f"  {Fore.GREEN}✅ {symbol} detuvo sus operaciones y confirmó guardado seguro.")
                        self.logger.info(f"Confirmación de guardado de datos recibida para {symbol}.")
            if len(confirmed_symbols) == len(SYMBOLS_TO_CAPTURE):
                print(f"\n{Fore.GREEN}✅ Todos los capturadores finalizaron de manera consistente.")
                self.logger.info("Todos los capturadores activos confirmaron un apagado limpio.")
                break
            time.sleep(0.5)
            
        missing_symbols = set(SYMBOLS_TO_CAPTURE) - confirmed_symbols
        if missing_symbols:
            self.logger.warning(f"⚠️ {len(missing_symbols)} capturadores no confirmaron dentro de {SHUTDOWN_GRACEFUL_WAIT}s: {missing_symbols}")
            for symbol in missing_symbols:
                info = self.processes.get(symbol, {})
                proc = info.get("process")
                if proc and proc.poll() is None:
                    try:
                        print(f"  {Fore.RED}⚡ Forzando terminación de {symbol} (PID: {proc.pid})...")
                        proc.terminate()
                        proc.wait(timeout=3)
                    except subprocess.TimeoutExpired: proc.kill()
                        
        self._clean_stale_signals()
        self.processes.clear()
        print(f"\n{Fore.GREEN}✅ Cierre de pipeline de captura completado.")

    def _fallback_force_kill(self):
        for symbol, info in self.processes.items():
            proc = info.get("process")
            if proc and proc.poll() is None:
                try: proc.kill()
                except Exception: pass

    def _capture_loop(self):
        last_health_check = time.time()
        last_visual_refresh = 0.0
        refresh_interval = 2.0
        while self.running:
            try:
                ahora = time.time()
                if ahora - last_visual_refresh >= refresh_interval:
                    self.print_status()
                    last_visual_refresh = ahora
                
                if not is_market_open():
                    self.market_open = False
                    print(f"\n{Fore.YELLOW}🔴 Mercado cerrado — Deteniendo pipeline de forma controlada...")
                    self.logger.info("🔴 Cierre de mercado. Deteniendo capturadores de forma controlada.")
                    self.shutdown_all_graceful()
                    return
                    
                dead_processes = self.check_dead_processes()
                for symbol, exit_code in dead_processes: self.restart_dead_process(symbol, exit_code)
                    
                if ahora - last_health_check > 120:
                    zombies = self.check_zombie_processes()
                    for symbol, edad, info in zombies:
                        session = self._current_session.get(symbol, self._get_current_session())
                        umbral_critico = get_session_umbral_critico(session)
                        if edad > umbral_critico * 1.5:
                            msg = f"Zombie detectado: {symbol} inactivo por {edad/60:.1f} minutos en sesión activa. Intentando apagado ordenado..."
                            self.logger.warning(msg)
                            send_telegram_alert(f"🚨 <b>{msg}</b>")
                            symbol_signal_file = STATE_DIR / f"{symbol}.shutdown_signal"
                            try:
                                with open(symbol_signal_file, "w", encoding="utf-8") as f:
                                    f.write(f"zombie_shutdown_initiated_at={time.time()}\n")
                            except Exception as e: self.logger.error(f"Fallo al escribir señal individual para {symbol}: {e}")
                                
                            confirmed = False
                            start_wait = time.time()
                            ok_file = STATE_DIR / f"{symbol}.shutdown_ok"
                            while time.time() - start_wait < 5.0:
                                if ok_file.exists(): confirmed = True; break
                                time.sleep(0.1)
                                
                            if not confirmed:
                                self.logger.warning(f"⚠️ Zombie {symbol} no respondió a la señal en disco. Aplicando fuerza de detención...")
                                if info.get('process'):
                                    info['process'].terminate()
                                    try: info['process'].wait(timeout=3)
                                    except subprocess.TimeoutExpired: info['process'].kill()
                                        
                            if symbol_signal_file.exists():
                                try: symbol_signal_file.unlink()
                                except Exception: pass
                            if ok_file.exists():
                                try: ok_file.unlink()
                                except Exception: pass
                            self.start_capturer_smart(symbol)
                    last_health_check = ahora
                    
                if self.kb_available:
                    import msvcrt
                    if msvcrt.kbhit():
                        key = msvcrt.getch().decode('utf-8', errors='ignore').upper()
                        if key == 'Q': self.running = False; break
                        elif key == 'R': self.restart_all(); time.sleep(3)
                        elif key == 'L': print(f"\n{Fore.CYAN}📂 Directorio centralizado de logs: {LOGS_UNIFIED_DIR}{Fore.RESET}"); time.sleep(2)
            except KeyboardInterrupt:
                self.running = False
                break
            except Exception as e:
                self.logger.error(f"Error en bucle de supervisión general: {e}", exc_info=True)
                time.sleep(5)

    def run(self):
        from SYSTEM_CANONICAL.system_sessions import get_current_broker_time
        broker_time = get_current_broker_time()
        self.logger.info(f"🚀 Launcher iniciado de forma integrada a las {broker_time.strftime('%H:%M:%S')}")
        self.check_market_before_start()
        while self.running:
            if not self.market_open:
                if not self._wait_for_market(): break
            if self.running and self.market_open:
                self.start_all_capturers()
                self._capture_loop()
                self.shutdown_all_graceful()
                
        self.logger.info("💀 Supervisor finalizado de forma limpia.")
